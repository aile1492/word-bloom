## 2. 리텐션 전략 개요

### 2.1 리텐션 퍼널 목표

```
설치 → D0(첫 세션) → D1(40%) → D7(20%) → D30(10%)
```

| 기간 | 목표 리텐션 | 핵심 드라이버 |
|------|------------|--------------|
| D0 (첫 세션) | 100% | 튜토리얼 완료, 첫 클리어 보상 |
| D1 (1일차) | 40% | Daily Challenge, 출석 보상 Day 2 예고 |
| D3 (3일차) | 30% | 스트릭 보너스 +10% 달성, 첫 업적 해금 |
| D7 (7일차) | 20% | 출석 보상 7일 프리미엄, 스트릭 보너스 +25% |
| D14 (14일차) | 15% | 시즌 이벤트 참여, 컬렉션 진행 |
| D30 (30일차) | 10% | 스트릭 +100% 보너스, 업적 Gold 티어 도전 |

## 3. 일일 출석 보상 (Daily Login Rewards)

### 3.1 보상 구조 (7일 주기, 무한 반복)

| Day | 보상 | 설명 |
|-----|------|------|
| Day 1 | 30 코인 | 기본 보상 |
| Day 2 | 40 코인 | 소폭 증가 |
| Day 3 | 50 코인 | |
| Day 4 | 60 코인 | |
| Day 5 | 80 코인 | 중간 점프 |
| Day 6 | 100 코인 | |
| Day 7 | 200 코인 + 랜덤 힌트 1개 | 프리미엄 보상 (주간 최고) |

- **총 7일 코인**: 560 코인 + 힌트 1개
- **주기 완료 후**: Day 1부터 다시 시작 (무한 반복)
- **중간에 하루라도 빠지면**: Day 1부터 재시작

### 3.2 시간 기준

- **리셋 시간**: 로컬 시간 00:00 (자정)
- **수령 가능 기간**: 해당 일자 00:00 ~ 23:59:59
- **판정 기준**: `Time.get_date_string_from_system()` 기반 로컬 날짜 비교

- **앱 실행 시 자동 팝업** (하루 1회, 미수령 시에만)
- Day 7 보상은 특별 이펙트 (파티클 + 효과음)
- 다음 주기의 Day 1~7 보상도 미리 볼 수 있음 (동일 구조)

### 3.4 daily_reward_manager.gd Pseudocode

```gdscript
# daily_reward_manager.gd
# 일일 출석 보상 관리자
class_name DailyRewardManager
extends Node

signal reward_claimed(day: int, rewards: Dictionary)
signal cycle_reset()

# 7일 주기 보상 테이블
const REWARD_TABLE: Array[Dictionary] = [
    {"day": 1, "coins": 30, "hint": 0},
    {"day": 2, "coins": 40, "hint": 0},
    {"day": 3, "coins": 50, "hint": 0},
    {"day": 4, "coins": 60, "hint": 0},
    {"day": 5, "coins": 80, "hint": 0},
    {"day": 6, "coins": 100, "hint": 0},
    {"day": 7, "coins": 200, "hint": 1},  # 프리미엄 보상
]

const CYCLE_LENGTH: int = 7

var save_manager: SaveManager  # P02_03 세이브 시스템 참조

func _ready() -> void:
    save_manager = get_node("/root/SaveManager")


## 오늘 날짜 문자열 반환 (YYYY-MM-DD)
func _get_today() -> String:
    return Time.get_date_string_from_system()


## 날짜 문자열 → Unix timestamp 변환 (일 단위 비교용)
func _date_to_days(date_str: String) -> int:
    if date_str.is_empty():
        return 0
    var parts := date_str.split("-")
    var dict := {
        "year": int(parts[0]),
        "month": int(parts[1]),
        "day": int(parts[2]),
        "hour": 0, "minute": 0, "second": 0
    }
    var unix := Time.get_unix_time_from_datetime_dict(dict)
    return int(unix) / 86400  # 일 단위로 변환


## 보상 수령 가능 여부 확인
func can_claim_today() -> bool:
    var save_data := save_manager.get_data()
    var last_claim: String = save_data.get("daily_reward_last_claim", "")
    var today := _get_today()
    return last_claim != today


## 현재 주기의 몇 일차인지 반환 (1~7)
func get_current_day() -> int:
    var save_data := save_manager.get_data()
    return save_data.get("daily_reward_day", 1)


## 연속 출석 여부 확인 (하루라도 빠졌으면 리셋)
func _check_streak_broken() -> bool:
    var save_data := save_manager.get_data()
    var last_claim: String = save_data.get("daily_reward_last_claim", "")
    if last_claim.is_empty():
        return false  # 첫 접속
    var today_days := _date_to_days(_get_today())
    var last_days := _date_to_days(last_claim)
    var diff := today_days - last_days
    return diff > 1  # 1일 초과 차이 = 중간에 빠짐


## 보상 수령 처리
func claim_daily_reward() -> Dictionary:
    if not can_claim_today():
        return {"success": false, "reason": "already_claimed"}

    var save_data := save_manager.get_data()

    # 연속 출석 끊김 확인
    if _check_streak_broken():
        save_data["daily_reward_day"] = 1  # Day 1부터 재시작
        cycle_reset.emit()

    var current_day: int = save_data.get("daily_reward_day", 1)
    var reward: Dictionary = REWARD_TABLE[current_day - 1].duplicate()

    # 스트릭 보너스 적용 (StreakManager 연동)
    var streak_mgr := get_node("/root/StreakManager") as StreakManager
    var streak_bonus: float = streak_mgr.get_coin_bonus_multiplier()
    reward["coins"] = int(reward["coins"] * (1.0 + streak_bonus))
    reward["bonus_applied"] = streak_bonus

    # 보상 지급
    var currency_mgr := get_node("/root/CurrencyManager")
    currency_mgr.add_coins(reward["coins"])
    if reward["hint"] > 0:
        currency_mgr.add_hints(reward["hint"])

    # 세이브 데이터 갱신
    var next_day: int = current_day + 1
    if next_day > CYCLE_LENGTH:
        next_day = 1  # 7일 완료 → 다시 Day 1
    save_data["daily_reward_day"] = next_day
    save_data["daily_reward_last_claim"] = _get_today()
    save_manager.save()

    reward["success"] = true
    reward["claimed_day"] = current_day
    reward_claimed.emit(current_day, reward)
    return reward


## 보상 캘린더 데이터 (UI용)
func get_calendar_data() -> Array[Dictionary]:
    var save_data := save_manager.get_data()
    var current_day: int = save_data.get("daily_reward_day", 1)
    var claimed_today: bool = not can_claim_today()
    var result: Array[Dictionary] = []

    for i in range(CYCLE_LENGTH):
        var day_num: int = i + 1
        var entry: Dictionary = REWARD_TABLE[i].duplicate()
        if day_num < current_day:
            entry["status"] = "claimed"     # 이미 수령
        elif day_num == current_day:
            entry["status"] = "today_claimed" if claimed_today else "today_available"
        else:
            entry["status"] = "locked"      # 미래 보상
        result.append(entry)

    return result
```

**GDScript 의사코드**:
```gdscript
# timer_widget.gd
extends PanelContainer

@export var cooldown_hours: float = 4.0
@export var reward_coins_min: int = 30
@export var reward_coins_max: int = 30
@export var save_key: String = "free_coin_last_claim"

@onready var timer_label: Label = $TimerLabel
@onready var claim_button: Button = $ClaimButton

func _process(delta: float) -> void:
    var last_claim: int = SaveManager.get_value(save_key, 0)
    var elapsed: float = Time.get_unix_time_from_system() - last_claim
    var cooldown_sec: float = cooldown_hours * 3600.0
    var remaining: float = cooldown_sec - elapsed

    if remaining <= 0:
        timer_label.visible = false
        claim_button.visible = true
    else:
        timer_label.visible = true
        claim_button.visible = false
        var hours: int = int(remaining / 3600)
        var minutes: int = int(fmod(remaining, 3600) / 60)
        timer_label.text = "%dh %dm" % [hours, minutes]

func _on_claim_pressed() -> void:
    var reward: int = randi_range(reward_coins_min, reward_coins_max)
    SaveManager.add_coins(reward)
    SaveManager.set_value(save_key, int(Time.get_unix_time_from_system()))
```

## 4. 스트릭 시스템 (Streak)

### 4.1 개요

플레이어가 매일 연속으로 게임을 플레이하면 코인 보너스 배율이 점차 증가하는 시스템. 스트릭이 길수록 보상이 커지며, 스트릭 보호막을 통해 하루 실수를 방어할 수 있다.

### 4.2 스트릭 보너스 테이블

| 연속 일수 | 코인 보너스 | 아이콘 | 설명 |
|-----------|-----------|--------|------|
| 1~2일 | +0% | - | 기본 |
| 3~6일 | +10% | 불꽃 1단계 | "불꽃 시작!" |
| 7~13일 | +25% | 불꽃 2단계 | "일주일 달성!" |
| 14~29일 | +50% | 불꽃 3단계 | "2주 연속!" |
| 30일+ | +100% | 불꽃 MAX | "한 달 마스터!" |

- 보너스는 **모든 코인 획득에 적용** (스테이지 클리어, 출석 보상, 이벤트 보상)
- 보너스 적용 시 UI에 "+XX% 스트릭 보너스!" 표시

### 4.3 스트릭 판정 기준

- **스트릭 유지 조건**: 하루에 최소 1개 스테이지 클리어 또는 Daily Challenge 참여
- **판정 시점**: 로컬 시간 00:00 기준 날짜 변경 시
- **스트릭 시작**: 첫 플레이 시 streak_count = 1

### 4.4 스트릭 보호막 (Streak Shield)

| 항목 | 내용 |
|------|------|
| **효과** | 1회 사용 시 하루 미접속해도 스트릭 유지 |
| **최대 보유** | 3개 |
| **자동 사용** | 자정 판정 시 스트릭 끊김 감지되면 자동 소모 |
| **획득 경로** | 7일차 출석 보상 (1개), 코인 구매 (50코인), 업적 보상 |
| **알림** | 보호막 사용 시 다음 접속 때 "보호막이 스트릭을 지켰어요!" 팝업 |

### 4.5 스트릭 끊김 시 위로 보상

| 끊긴 스트릭 | 위로 보상 | 비고 |
|------------|----------|------|
| 1~2일 | 0 코인 | 위로 보상 없음 |
| 3~9일 | 끊긴 일수 x 5 코인 | 15~45 코인 |
| 10~29일 | 끊긴 일수 x 5 코인 | 50~145 코인 |
| 30일+ | 150 코인 (최대) | 상한선 적용 |

- 위로 보상과 함께 "아쉽지만 다시 시작해봐요!" 메시지 표시
- 끊긴 스트릭 기록은 통계에 "최고 스트릭"으로 저장

### 4.6 streak_manager.gd Pseudocode

```gdscript
# streak_manager.gd
# 스트릭 시스템 관리자
class_name StreakManager
extends Node

signal streak_updated(count: int, bonus: float)
signal streak_broken(old_count: int, consolation_coins: int)
signal streak_shield_used(remaining: int)
signal streak_milestone(milestone: int)

const BONUS_TABLE: Array[Dictionary] = [
    {"min_days": 1, "max_days": 2, "bonus": 0.0},
    {"min_days": 3, "max_days": 6, "bonus": 0.10},
    {"min_days": 7, "max_days": 13, "bonus": 0.25},
    {"min_days": 14, "max_days": 29, "bonus": 0.50},
    {"min_days": 30, "max_days": 9999, "bonus": 1.0},
]

const MAX_SHIELDS: int = 3
const CONSOLATION_PER_DAY: int = 5
const CONSOLATION_MAX: int = 150
const MILESTONES: Array[int] = [3, 7, 14, 30, 50, 100]

var save_manager: SaveManager

func _ready() -> void:
    save_manager = get_node("/root/SaveManager")


func _get_today() -> String:
    return Time.get_date_string_from_system()


func _date_to_days(date_str: String) -> int:
    if date_str.is_empty():
        return 0
    var parts := date_str.split("-")
    var dict := {
        "year": int(parts[0]), "month": int(parts[1]), "day": int(parts[2]),
        "hour": 0, "minute": 0, "second": 0
    }
    return int(Time.get_unix_time_from_datetime_dict(dict)) / 86400


## 현재 스트릭 보너스 배율 반환 (0.0 ~ 1.0)
func get_coin_bonus_multiplier() -> float:
    var streak: int = get_streak_count()
    for entry in BONUS_TABLE:
        if streak >= entry["min_days"] and streak <= entry["max_days"]:
            return entry["bonus"]
    return 0.0

    save_manager.save()


## 앱 실행 시 호출 - 스트릭 상태 갱신
func check_and_update_streak() -> Dictionary:
    var save_data := save_manager.get_data()
    var last_play: String = save_data.get("streak_last_play", "")
    var today := _get_today()
    var result := {"action": "none"}

    if last_play.is_empty():
        # 첫 플레이 - 아직 스트릭 시작 아님
        return result

    var today_days := _date_to_days(today)
    var last_days := _date_to_days(last_play)
    var diff := today_days - last_days

    if diff == 0:
        # 오늘 이미 플레이함
        result["action"] = "already_played"
    elif diff == 1:
        # 어제 플레이함 - 스트릭 유지 중 (오늘 플레이하면 갱신)
        result["action"] = "streak_continuing"
    elif diff >= 2:
        # 하루 이상 빠짐 - 보호막 확인
        var missed_days := diff - 1  # 빠진 일수
        var shields: int = save_data.get("streak_shields", 0)

        if shields >= missed_days:
            # 보호막으로 커버 가능
            save_data["streak_shields"] = shields - missed_days
            save_manager.save()
            streak_shield_used.emit(save_data["streak_shields"])
            result["action"] = "shield_used"
            result["shields_used"] = missed_days
            result["shields_remaining"] = save_data["streak_shields"]
        else:
            # 스트릭 끊김
            var old_streak: int = save_data.get("streak_count", 0)
            var consolation: int = _calc_consolation(old_streak)

            # 최고 스트릭 기록 갱신
            var best: int = save_data.get("stats", {}).get("best_streak", 0)
            if old_streak > best:
                if not save_data.has("stats"):
                    save_data["stats"] = {}
                save_data["stats"]["best_streak"] = old_streak

            # 위로 보상 지급
            if consolation > 0:
                var currency_mgr := get_node("/root/CurrencyManager")
                currency_mgr.add_coins(consolation)

            # 스트릭 리셋
            save_data["streak_count"] = 0
            save_manager.save()

            streak_broken.emit(old_streak, consolation)
            result["action"] = "streak_broken"
            result["old_streak"] = old_streak
            result["consolation_coins"] = consolation

    return result


## 위로 보상 계산
func _calc_consolation(streak: int) -> int:
    if streak < 3:
        return 0
    return mini(streak * CONSOLATION_PER_DAY, CONSOLATION_MAX)


## 플레이 완료 시 호출 - 스트릭 갱신
func record_play() -> void:
    var save_data := save_manager.get_data()
    var last_play: String = save_data.get("streak_last_play", "")
    var today := _get_today()

    if last_play == today:
        return  # 오늘 이미 기록됨

    var today_days := _date_to_days(today)
    var last_days := _date_to_days(last_play)
    var diff := today_days - last_days

    if last_play.is_empty() or diff >= 2:
        # 첫 플레이 또는 보호막으로 유지된 후 플레이
        if save_data.get("streak_count", 0) == 0:
            save_data["streak_count"] = 1
    elif diff == 1:
        # 연속 플레이 - 스트릭 증가
        save_data["streak_count"] = save_data.get("streak_count", 0) + 1

    save_data["streak_last_play"] = today
    save_manager.save()

    var current_streak: int = save_data["streak_count"]
    var bonus: float = get_coin_bonus_multiplier()
    streak_updated.emit(current_streak, bonus)

    # 마일스톤 체크
    if current_streak in MILESTONES:
        streak_milestone.emit(current_streak)


## 스트릭 표시 데이터 (UI용)
func get_display_data() -> Dictionary:
    return {
        "streak_count": get_streak_count(),
        "bonus_multiplier": get_coin_bonus_multiplier(),
        "shields": get_shield_count(),
        "max_shields": MAX_SHIELDS,
        "next_milestone": _get_next_milestone(),
    }


func _get_next_milestone() -> int:
    var current: int = get_streak_count()
    for m in MILESTONES:
        if m > current:
            return m
    return -1  # 모든 마일스톤 달성
```

```gdscript
## 단어 발견 등록
func register_word(word: String, lang: String, theme: String, is_bonus: bool = false) -> void:
    var save_data := save_manager.get_data()
    if not save_data.has("collection_words"):
        save_data["collection_words"] = {}
    if not save_data["collection_words"].has(lang):
        save_data["collection_words"][lang] = {}
    if not save_data["collection_words"][lang].has(theme):
        save_data["collection_words"][lang][theme] = []

    var theme_words: Array = save_data["collection_words"][lang][theme]
    if word not in theme_words:
        theme_words.append(word)
        save_data["collection_words"][lang][theme] = theme_words
        save_manager.save()
        word_collected.emit(word, lang, theme)

        # 테마 완성 체크
        _check_theme_completion(lang, theme)
        # 마일스톤 체크
        _check_milestones()

        var ach_mgr := get_node("/root/AchievementManager") as AchievementManager
        ach_mgr.on_stat_updated("total_words_found", get_total_word_count())
        if is_bonus:
            ach_mgr.on_stat_updated("bonus_words_found", get_bonus_word_count())


## 총 발견 단어 수
func get_total_word_count() -> int:
    var count: int = 0
    var save_data := save_manager.get_data()
    var collection: Dictionary = save_data.get("collection_words", {})
    for lang in collection:
        for theme in collection[lang]:
            count += collection[lang][theme].size()
    return count


## 보너스 워드 발견 수
func get_bonus_word_count() -> int:
    var save_data := save_manager.get_data()
    return save_data.get("stats", {}).get("bonus_words_found", 0)


## 테마별 완성도 반환
func get_theme_progress(lang: String, theme: String) -> Dictionary:
    var save_data := save_manager.get_data()
    var found: Array = save_data.get("collection_words", {}) \
        .get(lang, {}).get(theme, [])
    var total_words: Array = word_database.get(lang, {}).get(theme, [])
    return {
        "found": found.size(),
        "total": total_words.size(),
        "percentage": float(found.size()) / max(total_words.size(), 1) * 100.0,
        "found_words": found,
        "missing_count": total_words.size() - found.size(),
    }


## 테마 완성 체크
func _check_theme_completion(lang: String, theme: String) -> void:
    var progress := get_theme_progress(lang, theme)
    if progress["found"] >= progress["total"] and progress["total"] > 0:
        theme_completed.emit(lang, theme)


## 마일스톤 체크
func _check_milestones() -> void:
    var save_data := save_manager.get_data()
    var claimed: Array = save_data.get("collection_milestones_claimed", [])
    var total_words: int = get_total_word_count()

    for ms in MILESTONES:
        if ms["id"] in claimed:
            continue
        var check_value: int = 0
        match ms["type"]:
            "total_words": check_value = total_words
            "bonus_words": check_value = get_bonus_word_count()

        if check_value >= ms["target"]:
            # 마일스톤 달성
            claimed.append(ms["id"])
            save_data["collection_milestones_claimed"] = claimed
            var currency_mgr := get_node("/root/CurrencyManager")
            currency_mgr.add_coins(ms["coins"])
            save_manager.save()
            milestone_reached.emit(ms["id"], ms)


## 도감 데이터 (UI용)
func get_collection_display(lang: String) -> Array[Dictionary]:
    var result: Array[Dictionary] = []
    var themes: Dictionary = word_database.get(lang, {})
    for theme_name in themes:
        var progress := get_theme_progress(lang, theme_name)
        result.append({
            "theme": theme_name,
            "found": progress["found"],
            "total": progress["total"],
            "percentage": progress["percentage"],
        })
    # 완성도 순 정렬
    result.sort_custom(func(a, b): return a["percentage"] > b["percentage"])
    return result
```

## 7. 시즌 이벤트

### 7.1 개요

계절 및 기념일에 맞춘 한정 이벤트를 운영하여 신선한 콘텐츠를 제공하고 FOMO(Fear Of Missing Out)를 통해 접속 동기를 부여한다. 모든 이벤트는 로컬 날짜 기반으로 활성화되며, 앱 업데이트를 통해 이벤트 콘텐츠를 추가한다.

```gdscript
func claim_reward(reward_id: String) -> Dictionary:
    if _current_event.is_empty():
        return {"success": false, "reason": "no_active_event"}

    var save_data := save_manager.get_data()
    var event_id: String = _current_event["id"]
    var event_save: Dictionary = save_data.get("event_data", {}).get(event_id, {})
    var claimed: Array = event_save.get("claimed", [])

    if reward_id in claimed:
        return {"success": false, "reason": "already_claimed"}

    # 보상 찾기
    var reward: Dictionary = {}
    for r in _current_event["rewards"]:
        if r["id"] == reward_id:
            reward = r
            break
    if reward.is_empty():
        return {"success": false, "reason": "reward_not_found"}

    # 포인트 충분한지 확인
    var current_points: int = event_save.get("points", 0)
    if current_points < reward["cost"]:
        return {"success": false, "reason": "not_enough_points"}

    # 포인트 차감 + 보상 지급
    event_save["points"] = current_points - reward["cost"]
    claimed.append(reward_id)
    event_save["claimed"] = claimed
    save_data["event_data"][event_id] = event_save
    save_manager.save()

    _grant_reward(reward)
    event_reward_claimed.emit(event_id, reward_id)
    return {"success": true, "reward": reward}

            profile_mgr.unlock_avatar(reward["item_id"])
        "frame":
            var profile_mgr := get_node("/root/ProfileManager")
            profile_mgr.unlock_frame(reward["item_id"])
        "theme":
            var profile_mgr := get_node("/root/ProfileManager")
            profile_mgr.unlock_theme(reward["item_id"])

            "id": r["id"],

            "cost": r["cost"],
            "type": r["type"],
            "is_claimed": r["id"] in claimed,
            "can_afford": current_points >= r["cost"],
        })
```

### 8.5 welcome_back_manager.gd Pseudocode

```gdscript
# welcome_back_manager.gd
# Welcome-Back 보너스 관리자
class_name WelcomeBackManager
extends Node

signal welcome_back_triggered(days_absent: int, rewards: Dictionary)
signal return_mission_completed(bonus_coins: int)

const COINS_PER_DAY: int = 20
const MAX_COIN_DAYS: int = 7
const LONG_ABSENCE_DAYS: int = 14
const SHORT_ABSENCE_DAYS: int = 3

const DDA_ADJUSTMENTS: Array[Dictionary] = [
    {"min_days": 3, "max_days": 6, "dda_offset": -1, "easy_stages": 2},
    {"min_days": 7, "max_days": 13, "dda_offset": -2, "easy_stages": 3},
    {"min_days": 14, "max_days": 9999, "dda_offset": -3, "easy_stages": 5},
]

var save_manager: SaveManager
var _remaining_easy_stages: int = 0
var _dda_offset: int = 0

func _ready() -> void:
    save_manager = get_node("/root/SaveManager")


func _get_today() -> String:
    return Time.get_date_string_from_system()


func _date_to_days(date_str: String) -> int:
    if date_str.is_empty():
        return 0
    var parts := date_str.split("-")
    var dict := {
        "year": int(parts[0]), "month": int(parts[1]), "day": int(parts[2]),
        "hour": 0, "minute": 0, "second": 0
    }
    return int(Time.get_unix_time_from_datetime_dict(dict)) / 86400


## 앱 실행 시 호출 - 미접속 기간 체크 후 보상 결정
func check_welcome_back() -> Dictionary:
    var save_data := save_manager.get_data()
    var last_login: String = save_data.get("welcome_back_last_login", "")
    var today := _get_today()

    # 마지막 로그인 기록 갱신
    save_data["welcome_back_last_login"] = today
    save_manager.save()

    if last_login.is_empty():
        return {"triggered": false, "reason": "first_login"}

    var today_days := _date_to_days(today)
    var last_days := _date_to_days(last_login)
    var days_absent: int = today_days - last_days

    if days_absent < SHORT_ABSENCE_DAYS:
        return {"triggered": false, "reason": "recent_login", "days": days_absent}

    # Welcome-Back 보상 생성
    var rewards := _calculate_rewards(days_absent)

    # 보상 지급
    _grant_rewards(rewards)

    # DDA 조정 설정
    _setup_dda(days_absent)

    # 복귀 미션 설정 (14일+)
    if days_absent >= LONG_ABSENCE_DAYS:
        _setup_return_mission()

    welcome_back_triggered.emit(days_absent, rewards)
    return {"triggered": true, "days_absent": days_absent, "rewards": rewards}


## 보상 계산
func _calculate_rewards(days_absent: int) -> Dictionary:
    var coin_days: int = mini(days_absent, MAX_COIN_DAYS)
    var rewards := {
        "coins": coin_days * COINS_PER_DAY,
        "hints": 0,
        "streak_shields": 0,
        "daily_challenge_ticket": 1,  # 놓친 Daily 1회
        "return_mission": false,
    }

    if days_absent >= LONG_ABSENCE_DAYS:
        rewards["streak_shields"] = 1
        rewards["hints"] = 3
        rewards["return_mission"] = true  # 3일 내 5스테이지 → 200코인

    return rewards


## 보상 지급
func _grant_rewards(rewards: Dictionary) -> void:
    var currency_mgr := get_node("/root/CurrencyManager")
    currency_mgr.add_coins(rewards["coins"])
    if rewards["hints"] > 0:
        currency_mgr.add_hints(rewards["hints"])
    if rewards["streak_shields"] > 0:
        var streak_mgr := get_node("/root/StreakManager") as StreakManager
        streak_mgr.add_shield(rewards["streak_shields"])


## DDA 조정 설정
func _setup_dda(days_absent: int) -> void:
    for adj in DDA_ADJUSTMENTS:
        if days_absent >= adj["min_days"] and days_absent <= adj["max_days"]:
            _remaining_easy_stages = adj["easy_stages"]
            _dda_offset = adj["dda_offset"]
            break


## 현재 DDA 오프셋 반환 (DDA 시스템에서 호출)
func get_dda_offset() -> int:
    if _remaining_easy_stages > 0:
        return _dda_offset
    return 0


## 스테이지 완료 시 호출 - 이지 스테이지 카운트 감소
func on_stage_completed() -> void:
    if _remaining_easy_stages > 0:
        _remaining_easy_stages -= 1


## 복귀 미션 설정
func _setup_return_mission() -> void:
    var save_data := save_manager.get_data()
    save_data["return_mission"] = {
        "active": true,
        "target_clears": 5,
        "current_clears": 0,
        "deadline": _get_today(),  # 시작일 (3일 내)
        "reward_coins": 200,
    }
    save_manager.save()


## 복귀 미션 진행 업데이트
func update_return_mission() -> void:
    var save_data := save_manager.get_data()
    var mission: Dictionary = save_data.get("return_mission", {})
    if not mission.get("active", false):
        return

    mission["current_clears"] += 1
    if mission["current_clears"] >= mission["target_clears"]:
        # 미션 완료
        mission["active"] = false
        var currency_mgr := get_node("/root/CurrencyManager")
        currency_mgr.add_coins(mission["reward_coins"])
        return_mission_completed.emit(mission["reward_coins"])

    save_data["return_mission"] = mission
    save_manager.save()
```

```gdscript
extends Node
```

```gdscript
const MAX_LOG_ENTRIES: int = 1000  # 로컬 로그 최대 보관 수

var _session_stages_played: int = 0

    save_manager = get_node("/root/SaveManager")
    _start_session()


## 세션 시작 기록
func _start_session() -> void:
    _session_start_time = int(Time.get_unix_time_from_system())
    _session_stages_played = 0
    var save_data := save_manager.get_data()
    var last_login: String = save_data.get("welcome_back_last_login", "")
    var days_since: int = 0
    if not last_login.is_empty():

        var last_days := _date_to_days(last_login)
        days_since = today_days - last_days

    _log_event("session_start", {
        "timestamp": _session_start_time,
        "days_since_last": days_since,
    })


## 앱 종료/백그라운드 시 호출
func end_session() -> void:
    var duration: int = int(Time.get_unix_time_from_system()) - _session_start_time
    _log_event("session_end", {
        "duration_seconds": duration,
        "stages_played": _session_stages_played,
    })

    # 통계 갱신
    var save_data := save_manager.get_data()
    if not save_data.has("stats"):
        save_data["stats"] = {}
    var total_time: int = save_data["stats"].get("total_play_time_seconds", 0)
    save_data["stats"]["total_play_time_seconds"] = total_time + duration
    save_manager.save()


func _date_to_days(date_str: String) -> int:
    var parts := date_str.split("-")
    var dict := {
        "year": int(parts[0]), "month": int(parts[1]), "day": int(parts[2]),
        "hour": 0, "minute": 0, "second": 0
    }
    return int(Time.get_unix_time_from_datetime_dict(dict)) / 86400


## 범용 이벤트 로깅
func log_daily_login(day: int, streak: int) -> void:
    _log_event("daily_login", {
        "day_in_cycle": day,
        "streak_count": streak,
        "date": Time.get_date_string_from_system(),
    })

    })

    })

    })

        "rewards": rewards,

    _log_event("stage_cleared", {
        "mode": mode,
        "rank": rank,

func _log_event(event_id: String, data: Dictionary) -> void:
    var log_entry := {
        "event": event_id,
        "timestamp": int(Time.get_unix_time_from_system()),
        "data": data,
    }

    var logs: Array = _load_logs()
    logs.append(log_entry)

    # 최대 보관 수 초과 시 오래된 항목 제거
    while logs.size() > MAX_LOG_ENTRIES:
        logs.pop_front()

    _save_logs(logs)


func _load_logs() -> Array:
    if not FileAccess.file_exists(LOG_FILE_PATH):
        return []
    var file := FileAccess.open(LOG_FILE_PATH, FileAccess.READ)
    var json := JSON.new()
    var err := json.parse(file.get_as_text())
    if err == OK and json.data is Array:
        return json.data
    return []


func _save_logs(logs: Array) -> void:
    var file := FileAccess.open(LOG_FILE_PATH, FileAccess.WRITE)
    file.store_string(JSON.stringify(logs, "  "))


## 리텐션 통계 요약 (디버그/분석용)
func get_retention_summary() -> Dictionary:
    var logs: Array = _load_logs()
    var login_dates: Array[String] = []
    var total_sessions: int = 0
    var total_duration: int = 0

    for entry in logs:
        match entry["event"]:
            "daily_login":
                var date: String = entry["data"].get("date", "")
                if date not in login_dates:
                    login_dates.append(date)
            "session_start":
                total_sessions += 1
            "session_end":
                total_duration += entry["data"].get("duration_seconds", 0)

    return {
        "unique_login_days": login_dates.size(),
        "total_sessions": total_sessions,
        "avg_session_seconds": total_duration / maxi(total_sessions, 1),
        "total_play_time_seconds": total_duration,
    }
```

```
2. AnalyticsLogger._start_session()      # 세션 시작 기록
3. WelcomeBackManager.check_welcome_back() # 복귀 보상 체크
4. StreakManager.check_and_update_streak() # 스트릭 상태 갱신

7. AchievementManager (진행도 갱신)        # 업적 상태 갱신
8. NotificationManager (알림 재스케줄)     # 알림 스케줄 갱신
```

## 부록 B: 매니저 간 Signal 연결도

```
DailyRewardManager
  ├─→ reward_claimed → AnalyticsLogger.log_daily_login()
  └─→ reward_claimed → StreakManager (보너스 적용 확인)

StreakManager
  ├─→ streak_updated → UI (스트릭 표시 갱신)
  ├─→ streak_broken → AnalyticsLogger.log_streak_broken()
  ├─→ streak_broken → AchievementManager.on_stat_updated("best_streak")
  ├─→ streak_milestone → AchievementManager.check_progress("ACH_STREAK")
  └─→ streak_shield_used → UI (보호막 사용 알림)

AchievementManager
  ├─→ achievement_unlocked → AnalyticsLogger.log_achievement_unlocked()
  ├─→ achievement_unlocked → UI (업적 달성 팝업)
  └─→ achievement_unlocked → ProfileManager (해금 아이템 지급)

CollectionManager
  ├─→ word_collected → AchievementManager.on_stat_updated()
  ├─→ milestone_reached → UI (마일스톤 팝업)
  └─→ theme_completed → AchievementManager.on_stat_updated("themes_completed")

SeasonEventManager
  ├─→ event_started → UI (이벤트 배너 표시)
  ├─→ event_points_earned → AnalyticsLogger.log_event_participated()
  └─→ event_reward_claimed → ProfileManager.unlock_*()

WelcomeBackManager
  ├─→ welcome_back_triggered → AnalyticsLogger.log_returned_from_absence()
  ├─→ welcome_back_triggered → UI (환영 팝업)
  └─→ return_mission_completed → UI (미션 완료 팝업)

NotificationManager
  └─→ (앱 백그라운드 시) schedule_notifications()
```
