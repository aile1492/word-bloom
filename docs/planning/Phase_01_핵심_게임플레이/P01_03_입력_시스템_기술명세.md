## 1. InputMap 설정

프로젝트 설정(`Project > Project Settings > Input Map`)에 다음 Action을 등록한다. Godot 4.6의 내장 `ui_*` 액션은 기본값을 유지하되, 게임 전용 액션을 추가로 정의한다.

### 1.1 Action 정의 테이블

| Action 이름 | Key / Touch | 설명 |
|-------------|-------------|------|
| `ui_accept` | `Enter`, `Space`, Joypad `A` (Xbox), `Cross` (PS) | 확인 / 선택 (Godot 기본 내장) |
| `ui_cancel` | `Escape`, Joypad `B` (Xbox), `Circle` (PS) | 취소 / 뒤로가기 (Godot 기본 내장) |
| `touch_press` | `Mouse Left Button` | 마우스 좌클릭 (PC에서 터치 대용) |
| `ui_up` | `Arrow Up`, Joypad `D-pad Up`, `Left Stick Up` | 위쪽 이동 (Godot 기본 내장) |
| `ui_down` | `Arrow Down`, Joypad `D-pad Down`, `Left Stick Down` | 아래쪽 이동 (Godot 기본 내장) |
| `ui_left` | `Arrow Left`, Joypad `D-pad Left`, `Left Stick Left` | 왼쪽 이동 (Godot 기본 내장) |
| `ui_right` | `Arrow Right`, Joypad `D-pad Right`, `Left Stick Right` | 오른쪽 이동 (Godot 기본 내장) |
| `grid_select` | Joypad `A` (Xbox), `Cross` (PS) | TV 모드 격자 셀 선택/확인 (커스텀) |
| `grid_cancel` | Joypad `B` (Xbox), `Circle` (PS) | TV 모드 격자 선택 취소 (커스텀) |

### 1.2 InputMap 코드 등록 (project.godot 또는 런타임)

```gdscript
## input_setup.gd (Autoload 또는 Main에서 호출)
## 프로젝트 설정에서 미리 등록하는 것을 권장하지만,
## 런타임에도 추가 가능하다.

func _setup_custom_input_actions() -> void:
	# grid_select: Joypad A 버튼
	if not InputMap.has_action("grid_select"):
		InputMap.add_action("grid_select")
		var event_joypad := InputEventJoypadButton.new()
		event_joypad.button_index = JOY_BUTTON_A
		InputMap.action_add_event("grid_select", event_joypad)

	# grid_cancel: Joypad B 버튼
	if not InputMap.has_action("grid_cancel"):
		InputMap.add_action("grid_cancel")
		var event_joypad := InputEventJoypadButton.new()
		event_joypad.button_index = JOY_BUTTON_B
		InputMap.action_add_event("grid_cancel", event_joypad)
```

## 2. 플랫폼 감지

게임은 **모바일(터치)**, **PC(마우스)**, **TV(D-pad)** 세 가지 입력 모드를 지원한다. 입력 모드는 실행 환경에 따라 자동으로 결정되며, 런타임에도 전환될 수 있다.

### 2.1 입력 모드 정의

```gdscript
## input_mode.gd
## 입력 모드 열거형. 현재 활성화된 입력 장치에 따라 결정된다.
class_name InputMode

enum Type {
	TOUCH,   ## 모바일 터치스크린 (드래그 + 탭)
	MOUSE,   ## PC 마우스 (클릭 드래그)
	DPAD,    ## TV/콘솔 D-pad (포커스 내비게이션)
}
```

### 2.2 플랫폼 감지 로직

```gdscript
## layout_manager.gd
## 현재 플랫폼과 입력 모드를 감지하고 관리한다.
## Autoload 싱글톤으로 등록하여 전역 접근을 제공한다.
extends Node

## 현재 TV 모드 여부
var is_tv: bool = false

## 현재 입력 모드
var current_input_mode: InputMode.Type = InputMode.Type.TOUCH

## 입력 모드가 변경되었을 때 발행되는 signal
signal input_mode_changed(new_mode: InputMode.Type)


func _ready() -> void:
	_detect_platform()


## 플랫폼 자동 감지
func _detect_platform() -> void:
	# TV 감지: Android이면서 터치스크린이 없는 경우
	# Android TV, Fire TV, Google TV 등
	is_tv = (
		OS.has_feature("android")
		and not DisplayServer.is_touchscreen_available()
	)

	if is_tv:
		current_input_mode = InputMode.Type.DPAD
	elif OS.has_feature("mobile") or DisplayServer.is_touchscreen_available():
		current_input_mode = InputMode.Type.TOUCH
	else:
		current_input_mode = InputMode.Type.MOUSE

	print("LayoutManager: Platform detected - is_tv=%s, mode=%s" % [
		is_tv,
		InputMode.Type.keys()[current_input_mode]
	])


## 런타임 입력 모드 전환 감지.
## 게임패드가 연결/해제되거나 터치 이벤트가 발생하면 모드를 전환한다.
func _input(event: InputEvent) -> void:
	var new_mode := current_input_mode

	if event is InputEventScreenTouch or event is InputEventScreenDrag:
		new_mode = InputMode.Type.TOUCH
	elif event is InputEventMouseButton or event is InputEventMouseMotion:
		# 마우스 이벤트가 터치에서 에뮬레이트된 것이 아닌 경우만
		if not event is InputEventScreenTouch:
			new_mode = InputMode.Type.MOUSE
	elif event is InputEventJoypadButton or event is InputEventJoypadMotion:
		new_mode = InputMode.Type.DPAD

	if new_mode != current_input_mode:
		current_input_mode = new_mode
		input_mode_changed.emit(new_mode)
		print("LayoutManager: Input mode changed to %s" % [
			InputMode.Type.keys()[new_mode]
		])
```

### 2.3 플랫폼별 동작 차이

| 기능 | TOUCH (모바일) | MOUSE (PC) | DPAD (TV) |
|------|---------------|------------|-----------|
| 단어 선택 | 드래그 + 탭 | 클릭 드래그 | 포커스 이동 + 선택 버튼 |
| 실시간 하이라이트 | O | O | 포커스 셀만 하이라이트 |
| 셀 호버 효과 | X (불필요) | O (마우스 커서 위치) | O (포커스 셀) |
| 격자 외부 터치 | 선택 취소 | 선택 취소 | B 버튼으로 취소 |
| 최소 터치 영역 | 44px | 제한 없음 | 포커스 테두리 |

## 3. 터치 입력 처리

터치 및 마우스 입력은 `GridInputHandler`에서 통합 처리한다. **드래그 모드**와 **탭 모드** 두 가지 방식을 지원한다.

### 3.1 드래그 모드 상세

플레이어가 첫 번째 셀을 누른 상태에서 드래그하여 단어를 선택하는 방식이다.

```gdscript
## grid_input_handler.gd (드래그 모드 부분)

## 드래그 시작 - 터치/마우스를 누른 순간
func _start_drag(screen_pos: Vector2) -> void:
	var grid_pos := _screen_to_grid(screen_pos)

	# 격자 범위 내인지 확인
	if not _grid_data.is_in_bounds(grid_pos):
		return

	_state = State.DRAGGING
	_drag_start_cell = grid_pos
	_drag_start_screen = screen_pos
	_current_selected_cells.clear()
	_current_selected_cells.append(grid_pos)

	drag_started.emit(grid_pos)


## 드래그 업데이트 - 터치/마우스를 누른 상태로 이동하는 동안
func _update_drag(screen_pos: Vector2) -> void:
	if _state != State.DRAGGING:
		return

	# 현재 드래그 위치에서 방향 스냅
	var snapped_dir := DirectionSnapper.snap_direction(
		Vector2(_drag_start_screen),
		screen_pos
	)

	if snapped_dir == Vector2i.ZERO:
		return

	# 드래그 거리를 셀 단위로 계산
	var current_grid_pos := _screen_to_grid(screen_pos)
	var cell_dist := DirectionSnapper.cell_distance(_drag_start_cell, current_grid_pos)

	# 최소 1셀 이상 선택
	cell_dist = maxi(cell_dist, 1)

	# 스냅된 경로 계산
	var new_cells := DirectionSnapper.get_snapped_path(
		_drag_start_cell,
		snapped_dir,
		cell_dist + 1,  # +1: 시작 셀 포함
		_grid_data.width,
		_grid_data.height
	)

	# 이전 경로와 동일하면 갱신 불필요
	if new_cells == _current_selected_cells:
		return

	_current_selected_cells = new_cells
	drag_updated.emit(_current_selected_cells)


## 드래그 종료 - 터치/마우스를 뗀 순간
func _end_drag() -> void:
	if _state != State.DRAGGING:
		return

	_state = State.IDLE
	var final_cells := _current_selected_cells.duplicate()
	_current_selected_cells.clear()

	# 최소 2셀 이상 선택되어야 유효
	if final_cells.size() >= 2:
		drag_ended.emit(final_cells)
	else:
		# 한 셀만 선택된 경우 탭으로 처리
		if final_cells.size() == 1:
			_handle_tap(final_cells[0])
		drag_cancelled.emit()
```

### 3.2 탭 모드 상세

모바일에서 작은 셀을 드래그하기 어려운 경우를 위한 보조 입력 방식이다. 첫 번째 셀 탭 후 마지막 셀을 탭하면 사이의 모든 셀이 자동 선택된다.

```gdscript
## grid_input_handler.gd (탭 모드 부분)

## 탭 처리 - 셀을 짧게 터치(드래그 없이 뗀 경우)
func _handle_tap(grid_pos: Vector2i) -> void:
	if not _grid_data.is_in_bounds(grid_pos):
		return

	match _state:
		State.IDLE:
			# 첫 번째 셀 탭: TAP_FIRST 상태로 전환
			_state = State.TAP_FIRST
			_tap_first_cell = grid_pos
			cell_tapped.emit(grid_pos)

		State.TAP_FIRST:
			if grid_pos == _tap_first_cell:
				# 같은 셀을 다시 탭: 선택 취소
				_state = State.IDLE
				_tap_first_cell = Vector2i(-1, -1)
				tap_cancelled.emit()
			else:
				# 두 번째 셀 탭: 경로 계산 및 선택 완료
				_complete_tap_selection(grid_pos)


## 탭 선택 완료 - 두 셀 사이의 경로를 계산한다.
func _complete_tap_selection(second_cell: Vector2i) -> void:
	# 두 셀 사이의 방향이 유효한 8방향인지 확인
	if not DirectionSnapper.is_valid_direction(_tap_first_cell, second_cell):
		# 유효하지 않은 방향이면 무시 (사선이 45도가 아닌 경우)
		tap_invalid.emit(_tap_first_cell, second_cell)
		return

	# 방향 및 거리 계산
	var delta := second_cell - _tap_first_cell
	var direction := Vector2i(signi(delta.x), signi(delta.y))
	var distance := DirectionSnapper.cell_distance(_tap_first_cell, second_cell)

	# 경로 생성 (시작 셀 ~ 끝 셀)
	var cells: Array = []  # Array[Vector2i]
	for i in range(distance + 1):
		var pos := _tap_first_cell + direction * i
		if _grid_data.is_in_bounds(pos):
			cells.append(pos)

	# 상태 초기화
	_state = State.IDLE
	_tap_first_cell = Vector2i(-1, -1)

	# 결과 전달 (드래그 종료와 동일한 signal 사용)
	if cells.size() >= 2:
		drag_ended.emit(cells)
```

### 3.3 screen_to_grid 좌표 변환

화면(뷰포트) 좌표를 격자 셀 좌표(`Vector2i`)로 변환한다.

```gdscript
## grid_input_handler.gd

## GridBoard 컨테이너 참조
var _grid_board: CenterContainer = null

## 셀 크기 (px)
var _cell_size: float = 60.0

## 셀 간 간격 (px)
var _cell_gap: float = 2.0


## 화면 좌표를 격자 셀 좌표로 변환한다.
##
## 변환 과정:
## 1. 화면 좌표 → GridContainer 로컬 좌표 (부모 컨테이너의 오프셋 제거)
## 2. 로컬 좌표 → 격자 열/행 인덱스 (셀 크기 + 간격으로 나누기)
##
## [매개변수]
##   screen_pos: 화면(뷰포트) 기준 좌표
##
## [반환값]
##   격자 셀 좌표 (col, row). 범위 밖이면 음수 또는 격자 크기 초과 가능.
func _screen_to_grid(screen_pos: Vector2) -> Vector2i:
	# GridContainer의 글로벌 위치를 기준으로 로컬 좌표 계산
	var grid_container: GridContainer = _grid_board.get_node("GridContainer")
	var grid_global_pos: Vector2 = grid_container.global_position

	var local_x: float = screen_pos.x - grid_global_pos.x
	var local_y: float = screen_pos.y - grid_global_pos.y

	# 셀 크기 + 간격 단위로 열/행 계산
	var step: float = _cell_size + _cell_gap
	var col: int = int(local_x / step)
	var row: int = int(local_y / step)

	return Vector2i(col, row)


## 격자 셀 좌표를 화면 좌표(셀 중심)로 변환한다.
## 드래그 라인 렌더링 등에 사용한다.
func _grid_to_screen(grid_pos: Vector2i) -> Vector2:
	var grid_container: GridContainer = _grid_board.get_node("GridContainer")
	var grid_global_pos: Vector2 = grid_container.global_position

	var step: float = _cell_size + _cell_gap
	var center_x: float = grid_global_pos.x + grid_pos.x * step + _cell_size / 2.0
	var center_y: float = grid_global_pos.y + grid_pos.y * step + _cell_size / 2.0

	return Vector2(center_x, center_y)
```

### 3.4 이벤트 처리 통합

```gdscript
## grid_input_handler.gd - 이벤트 통합 처리
## 터치와 마우스 이벤트를 통합하여 처리한다.
## Godot 4.6에서는 InputEventScreenTouch/ScreenDrag가 터치 전용이고,
## InputEventMouseButton/MouseMotion이 마우스 전용이다.

func _input(event: InputEvent) -> void:
	# TV 모드에서는 터치/마우스 입력 무시
	if LayoutManager.current_input_mode == InputMode.Type.DPAD:
		return

	# === 터치 입력 ===
	if event is InputEventScreenTouch:
		if event.pressed:
			_start_drag(event.position)
		else:
			_end_drag()

	elif event is InputEventScreenDrag:
		_update_drag(event.position)

	# === 마우스 입력 (PC) ===
	elif event is InputEventMouseButton:
		if event.button_index == MOUSE_BUTTON_LEFT:
			if event.pressed:
				_start_drag(event.position)
			else:
				_end_drag()

	elif event is InputEventMouseMotion:
		if Input.is_mouse_button_pressed(MOUSE_BUTTON_LEFT):
			_update_drag(event.position)
		else:
			# 호버 효과 (마우스 모드에서만)
			_update_hover(event.position)
```

## 4. D-pad 포커스 내비게이션 (TV mode)

Android TV, Fire TV 등 리모컨/게임패드 환경에서는 터치 대신 **D-pad 포커스 이동**으로 셀을 선택한다. Godot의 내장 Focus 시스템을 활용한다.

### 4.1 Button 기반 포커스

각 `LetterCell`을 `Button` 노드(또는 `BaseButton` 파생)로 구현하여 Godot 내장 포커스 시스템을 활용한다.

```gdscript
## letter_cell.gd
## LetterCell은 PanelContainer를 상속하지만,
## TV 모드에서 포커스를 받을 수 있도록 focus_mode를 설정한다.
extends PanelContainer

## 셀의 격자 좌표
var grid_pos: Vector2i = Vector2i.ZERO

## 표시할 글자
var letter: String = ""

@onready var letter_label: Label = $MarginContainer/Label
@onready var highlight_overlay: ColorRect = $ColorRect


func _ready() -> void:
	# 모든 플랫폼에서 포커스 수신 가능 (TV에서 D-pad 내비게이션용)
	focus_mode = Control.FOCUS_ALL

	# 포커스 진입/이탈 시각 효과
	focus_entered.connect(_on_focus_entered)
	focus_exited.connect(_on_focus_exited)


## 셀 초기화
func setup(pos: Vector2i, char: String) -> void:
	grid_pos = pos
	letter = char
	letter_label.text = char


## D-pad로 포커스가 들어왔을 때
func _on_focus_entered() -> void:
	if LayoutManager.current_input_mode == InputMode.Type.DPAD:
		_set_visual_state(VisualState.HOVER)


## D-pad 포커스가 나갔을 때
func _on_focus_exited() -> void:
	if LayoutManager.current_input_mode == InputMode.Type.DPAD:
		_set_visual_state(VisualState.IDLE)
```

### 4.2 커스텀 포커스 순서

기본 포커스 순서는 노드 트리 순서를 따르지만, 격자에서는 **2D 그리드 탐색**이 필요하다. 각 셀의 상하좌우 이웃 셀 경로를 명시적으로 설정한다.

```gdscript
## grid_board.gd 내부
## TV 모드용 포커스 이웃 설정.
## 각 셀에 상하좌우 이웃 셀의 NodePath를 지정하여
## D-pad 입력 시 올바른 방향으로 포커스가 이동하도록 한다.

func _setup_focus_neighbors() -> void:
	var width: int = _grid_data.width
	var height: int = _grid_data.height

	for y in range(height):
		for x in range(width):
			var index: int = y * width + x
			var cell: Control = _cells[index]

			# 왼쪽 이웃
			if x > 0:
				var left_index: int = y * width + (x - 1)
				cell.focus_neighbor_left = _cells[left_index].get_path()
			else:
				# 격자 왼쪽 끝: 포커스 이동 차단 (자기 자신으로 설정)
				cell.focus_neighbor_left = cell.get_path()

			# 오른쪽 이웃
			if x < width - 1:
				var right_index: int = y * width + (x + 1)
				cell.focus_neighbor_right = _cells[right_index].get_path()
			else:
				cell.focus_neighbor_right = cell.get_path()

			# 위쪽 이웃
			if y > 0:
				var top_index: int = (y - 1) * width + x
				cell.focus_neighbor_top = _cells[top_index].get_path()
			else:
				cell.focus_neighbor_top = cell.get_path()

			# 아래쪽 이웃
			if y < height - 1:
				var bottom_index: int = (y + 1) * width + x
				cell.focus_neighbor_bottom = _cells[bottom_index].get_path()
			else:
				cell.focus_neighbor_bottom = cell.get_path()

	# 초기 포커스를 격자 중앙 셀에 설정
	var center_index: int = (height / 2) * width + (width / 2)
	_cells[center_index].grab_focus()
```

### 4.3 D-pad 셀 선택 로직

TV 모드에서는 D-pad로 포커스를 이동한 후 `grid_select` 액션(A 버튼)으로 셀을 선택한다. 탭 모드와 동일한 "두 번 선택" 방식을 사용한다.

```gdscript
## grid_input_handler_tv.gd
## TV 모드 전용 입력 처리.
## D-pad 포커스 이동은 Godot 내장 시스템이 처리하고,
## 여기서는 A/B 버튼의 셀 선택/취소만 담당한다.
extends Node

signal word_selected(cells: Array)  # Array[Vector2i]
signal selection_cancelled

## 현재 선택 상태
var _first_cell: Vector2i = Vector2i(-1, -1)
var _is_selecting: bool = false

## 참조
var _grid_board: CenterContainer = null
var _grid_data: GridData = null


func _unhandled_input(event: InputEvent) -> void:
	if LayoutManager.current_input_mode != InputMode.Type.DPAD:
		return

	# A 버튼: 셀 선택
	if event.is_action_pressed("grid_select"):
		_on_select_pressed()
		get_viewport().set_input_as_handled()

	# B 버튼: 선택 취소
	elif event.is_action_pressed("grid_cancel"):
		_on_cancel_pressed()
		get_viewport().set_input_as_handled()


## A 버튼 누름: 셀 선택
func _on_select_pressed() -> void:
	# 현재 포커스된 셀 찾기
	var focused := get_viewport().gui_get_focus_owner()
	if focused == null or not focused.has_method("get_grid_pos"):
		return

	var grid_pos: Vector2i = focused.grid_pos

	if not _is_selecting:
		# 첫 번째 셀 선택
		_first_cell = grid_pos
		_is_selecting = true
		_highlight_first_cell(grid_pos)
	else:
		# 두 번째 셀 선택: 경로 계산
		if grid_pos == _first_cell:
			# 같은 셀 재선택: 취소
			_cancel_selection()
			return

		# 유효한 8방향인지 확인
		if not DirectionSnapper.is_valid_direction(_first_cell, grid_pos):
			# 유효하지 않은 방향: 무시 또는 피드백
			_show_invalid_direction_feedback()
			return

		# 경로 생성
		var delta := grid_pos - _first_cell
		var direction := Vector2i(signi(delta.x), signi(delta.y))
		var distance := DirectionSnapper.cell_distance(_first_cell, grid_pos)

		var cells: Array = []  # Array[Vector2i]
		for i in range(distance + 1):
			var pos := _first_cell + direction * i
			cells.append(pos)

		# 선택 완료
		_is_selecting = false
		_first_cell = Vector2i(-1, -1)
		word_selected.emit(cells)


## B 버튼 누름: 선택 취소
func _on_cancel_pressed() -> void:
	if _is_selecting:
		_cancel_selection()


## 선택 취소
func _cancel_selection() -> void:
	_is_selecting = false
	_clear_first_cell_highlight()
	_first_cell = Vector2i(-1, -1)
	selection_cancelled.emit()


## 첫 번째 셀 하이라이트
func _highlight_first_cell(pos: Vector2i) -> void:
	var cell := _grid_board.get_cell_at(pos)
	if cell:
		cell.set_visual_state(LetterCell.VisualState.DRAGGING)


## 첫 번째 셀 하이라이트 해제
func _clear_first_cell_highlight() -> void:
	if _first_cell != Vector2i(-1, -1):
		var cell := _grid_board.get_cell_at(_first_cell)
		if cell:
			cell.set_visual_state(LetterCell.VisualState.IDLE)


## 유효하지 않은 방향 피드백 (떨림 애니메이션 등)
func _show_invalid_direction_feedback() -> void:
	# 짧은 떨림 애니메이션으로 시각적 피드백 제공
	var focused := get_viewport().gui_get_focus_owner()
	if focused:
		var tween := create_tween()
		tween.tween_property(focused, "position:x", focused.position.x + 5, 0.05)
		tween.tween_property(focused, "position:x", focused.position.x - 5, 0.05)
		tween.tween_property(focused, "position:x", focused.position.x, 0.05)
```

```gdscript
## grid_input_handler.gd
## 격자 입력 처리의 메인 클래스.
## 터치/마우스 입력을 받아 셀 선택 결과를 signal로 전달한다.
## TV 모드는 별도의 grid_input_handler_tv.gd가 담당한다.
class_name GridInputHandler
extends Node


# ===== 상태 정의 =====

enum State {
	IDLE,        ## 대기 상태. 입력 대기 중.
	DRAGGING,    ## 드래그 중. 터치/마우스를 누른 상태로 이동 중.
	TAP_FIRST,   ## 탭 모드 첫 번째 셀 선택 완료. 두 번째 셀 대기 중.
}


# ===== Signals =====

## 드래그 시작됨 (시작 셀 좌표)
signal drag_started(start_cell: Vector2i)

## 드래그 업데이트됨 (현재 선택된 셀 경로)
signal drag_updated(cells: Array)  # Array[Vector2i]

## 드래그 종료됨 (최종 선택된 셀 경로). 매칭 판정 필요.
signal drag_ended(cells: Array)  # Array[Vector2i]

## 드래그 취소됨 (유효하지 않은 선택)
signal drag_cancelled

## 셀 탭됨 (탭 모드 첫 번째 셀 선택)
signal cell_tapped(pos: Vector2i)

## 탭 취소됨 (첫 번째 셀 재탭)
signal tap_cancelled

## 유효하지 않은 탭 방향
signal tap_invalid(first: Vector2i, second: Vector2i)

## 마우스 호버 셀 변경 (PC 모드 전용)
signal hover_changed(pos: Vector2i)  # (-1,-1) = 호버 없음


# ===== 상태 변수 =====

var _state: State = State.IDLE
var _drag_start_cell: Vector2i = Vector2i.ZERO
var _drag_start_screen: Vector2 = Vector2.ZERO
var _current_selected_cells: Array = []  # Array[Vector2i]
var _tap_first_cell: Vector2i = Vector2i(-1, -1)
var _hover_cell: Vector2i = Vector2i(-1, -1)


# ===== 참조 =====

var _grid_board: CenterContainer = null
var _grid_data: GridData = null
var _cell_size: float = 60.0
var _cell_gap: float = 2.0


# ===== 초기화 =====

## 입력 핸들러를 초기화한다.
func setup(grid_board: CenterContainer, grid_data: GridData) -> void:
	_grid_board = grid_board
	_grid_data = grid_data
	_cell_size = grid_board.get_cell_size()
	_cell_gap = GridLayout.CELL_GAP
	_state = State.IDLE


# ===== 이벤트 처리 =====

func _input(event: InputEvent) -> void:
	# TV 모드에서는 처리하지 않음
	if LayoutManager.current_input_mode == InputMode.Type.DPAD:
		return

	# 터치 이벤트
	if event is InputEventScreenTouch:
		if event.pressed:
			_start_drag(event.position)
		else:
			_end_drag()

	elif event is InputEventScreenDrag:
		_update_drag(event.position)

	# 마우스 이벤트
	elif event is InputEventMouseButton:
		if event.button_index == MOUSE_BUTTON_LEFT:
			if event.pressed:
				_start_drag(event.position)
			else:
				_end_drag()

	elif event is InputEventMouseMotion:
		if Input.is_mouse_button_pressed(MOUSE_BUTTON_LEFT):
			_update_drag(event.position)
		else:
			_update_hover(event.position)


# ===== 호버 처리 (PC 마우스 전용) =====

func _update_hover(screen_pos: Vector2) -> void:
	var grid_pos := _screen_to_grid(screen_pos)
	if not _grid_data.is_in_bounds(grid_pos):
		grid_pos = Vector2i(-1, -1)

	if grid_pos != _hover_cell:
		_hover_cell = grid_pos
		hover_changed.emit(_hover_cell)


# ===== 좌표 변환 =====

func _screen_to_grid(screen_pos: Vector2) -> Vector2i:
	var grid_container: GridContainer = _grid_board.get_node("GridContainer")
	var grid_global_pos: Vector2 = grid_container.global_position
	var local_x: float = screen_pos.x - grid_global_pos.x
	var local_y: float = screen_pos.y - grid_global_pos.y
	var step: float = _cell_size + _cell_gap
	return Vector2i(int(local_x / step), int(local_y / step))


func _grid_to_screen(grid_pos: Vector2i) -> Vector2:
	var grid_container: GridContainer = _grid_board.get_node("GridContainer")
	var grid_global_pos: Vector2 = grid_container.global_position
	var step: float = _cell_size + _cell_gap
	return Vector2(
		grid_global_pos.x + grid_pos.x * step + _cell_size / 2.0,
		grid_global_pos.y + grid_pos.y * step + _cell_size / 2.0
	)


# ===== 공개 메서드 =====

## 현재 상태를 반환한다.
func get_state() -> State:
	return _state


## 현재 선택된 셀 목록을 반환한다.
func get_selected_cells() -> Array:
	return _current_selected_cells.duplicate()


## 입력 상태를 강제 초기화한다 (스테이지 전환 등).
func reset() -> void:
	_state = State.IDLE
	_current_selected_cells.clear()
	_tap_first_cell = Vector2i(-1, -1)
	_hover_cell = Vector2i(-1, -1)
```

### 6.3 상태 전환 및 Tween 애니메이션

```gdscript
## letter_cell.gd

## 현재 활성화된 Tween (중복 방지용)
var _active_tween: Tween = null

## 시각적 상태를 변경하고 애니메이션을 재생한다.
func set_visual_state(new_state: VisualState, color_index: int = -1) -> void:
	# 이미 FOUND 상태인 셀은 IDLE/HOVER로 되돌리지 않음
	if _visual_state == VisualState.FOUND:
		if new_state == VisualState.IDLE or new_state == VisualState.HOVER:
			return

	_visual_state = new_state

	# 기존 Tween 중단
	if _active_tween and _active_tween.is_running():
		_active_tween.kill()

	# 목표 색상 결정
	var target_bg_color: Color
	var target_text_color: Color

	if new_state == VisualState.FOUND:
		_found_color_index = color_index
		var ci := clampi(color_index, 0, FOUND_COLORS.size() - 1)
		target_bg_color = FOUND_COLORS[ci]
		target_text_color = TEXT_COLORS[VisualState.FOUND]

		target_bg_color = STATE_COLORS[new_state]
		target_text_color = TEXT_COLORS.get(new_state, TEXT_COLORS[VisualState.IDLE])

	_active_tween = create_tween()
	_active_tween.set_parallel(true)

	# FOUND 상태 진입 시 팝업 애니메이션
	if new_state == VisualState.FOUND:
		_play_found_animation()

	pivot_offset = size / 2.0
```

```
    ├── _play_found_animation()     ← 정답 팝업 효과
```
