## 1. 게임 상태 머신 (FSM)

### 1.1 상태 정의

게임의 전체 생명주기를 5개의 상태로 관리한다. `GameController`가 Autoload로 등록되어 어디서든 접근 가능하다.

| 상태 | 설명 | 진입 조건 |
|------|------|-----------|
| **IDLE** | 게임 미진행 상태. 타이틀/메뉴 화면 | 앱 시작, 게임 종료 후 홈 복귀 |
| **PLAYING** | 퍼즐 플레이 중 | 스테이지 시작 |
| **PAUSED** | 일시 정지 (설정 팝업, 앱 백그라운드) | 설정 버튼, 앱 포커스 아웃 |
| **COMPLETED** | 스테이지 클리어 (모든 단어 발견) | 마지막 단어 발견 |
| **FAILED** | 스테이지 실패 (Time Attack 시간 초과) | 타이머 만료 |

### 1.2 상태 전이 다이어그램

```
                    start_stage()
            ┌───────────────────────┐
            │                       ▼
         ┌──────┐            ┌──────────┐
         │ IDLE │            │ PLAYING  │◄──────────┐
         └──┬───┘            └──┬───┬───┘           │
            ▲                   │   │            resume()
            │                   │   │               │
            │        pause()    │   │          ┌────────┐
            │         ┌─────────┘   │          │ PAUSED │
            │         │             │          └────────┘
            │         ▼             │
            │    ┌────────┐         │
            │    │ PAUSED │         │
            │    └────────┘         │
            │                       │
            │     all_found()       │   time_expired()
            │         ┌─────────────┘──────────┐
            │         ▼                        ▼
            │  ┌───────────┐           ┌────────┐
            │  │ COMPLETED │           │ FAILED │
            │  └─────┬─────┘           └───┬────┘
            │        │                     │
            │        │  go_home()          │ go_home()
            └────────┴─────────────────────┘
                     │
                     │ next_stage() / retry()
                     ▼
                  PLAYING (새 스테이지)
```

### 1.3 상태 전이 규칙 표

| 현재 상태 | 이벤트 | 다음 상태 | 조건 |
|-----------|--------|-----------|------|
| IDLE | `start_stage()` | PLAYING | - |
| PLAYING | `pause()` | PAUSED | - |
| PLAYING | `all_words_found()` | COMPLETED | `GridData.is_all_found() == true` |
| PLAYING | `time_expired()` | FAILED | Time Attack 모드 전용 |
| PAUSED | `resume()` | PLAYING | - |
| PAUSED | `go_home()` | IDLE | 진행 중인 게임 포기 |
| COMPLETED | `next_stage()` | PLAYING | Failed 상태에서는 불가 |
| COMPLETED | `retry()` | PLAYING | 동일 스테이지 재시작 |
| COMPLETED | `go_home()` | IDLE | - |
| FAILED | `retry()` | PLAYING | 동일 스테이지 재시작 |
| FAILED | `go_home()` | IDLE | - |

### 1.5 GameController 상태 머신 구현

```gdscript
## game_controller.gd
## 게임 전체 상태를 관리하는 Autoload 싱글톤.
## 포팅 원본: GameController.cs, GameEnums.cs
extends Node

class_name GameControllerClass

## --- 상태 Enum ---
enum GameState {
	IDLE,
	PLAYING,
	PAUSED,
	COMPLETED,
	FAILED,
}

## --- 게임 모드 Enum ---
enum GameMode {
	CLASSIC,
	TIME_ATTACK,
	DAILY_CHALLENGE,
	BONUS,
	MARATHON,
}

## --- 시그널 ---
signal game_started(stage: int, mode: GameMode)
signal game_paused()
signal game_resumed()
signal game_completed(result: GameResult)
signal game_failed(result: GameResult)
signal game_state_changed(old_state: GameState, new_state: GameState)

## --- 상태 변수 ---
var current_state: GameState = GameState.IDLE
var current_mode: GameMode = GameMode.CLASSIC
var current_stage: int = 1
var current_theme: String = ""

## --- 내부 변수 ---
var _play_start_time: float = 0.0
var _hint_count_used: int = 0
var _words_found_count: int = 0
var _total_words: int = 0
var _is_combo: bool = false


## 상태를 전이한다. 유효하지 않은 전이는 경고를 출력하고 무시한다.
func _change_state(new_state: GameState) -> void:
	if current_state == new_state:
		return

	var old_state := current_state

	# 전이 유효성 검사
	if not _is_valid_transition(old_state, new_state):
		push_warning(
			"GameController: 유효하지 않은 상태 전이 %s → %s" % [
				GameState.keys()[old_state],
				GameState.keys()[new_state]
			]
		)
		return

	current_state = new_state
	game_state_changed.emit(old_state, new_state)

	if OS.is_debug_build():
		print("GameController: %s → %s" % [
			GameState.keys()[old_state],
			GameState.keys()[new_state]
		])


## 전이 유효성 검사 테이블
func _is_valid_transition(from: GameState, to: GameState) -> bool:
	match from:
		GameState.IDLE:
			return to == GameState.PLAYING
		GameState.PLAYING:
			return to in [GameState.PAUSED, GameState.COMPLETED, GameState.FAILED]
		GameState.PAUSED:
			return to in [GameState.PLAYING, GameState.IDLE]
		GameState.COMPLETED:
			return to in [GameState.PLAYING, GameState.IDLE]
		GameState.FAILED:
			return to in [GameState.PLAYING, GameState.IDLE]
	return false


## --- 공개 API ---

## 스테이지를 시작한다.
func start_stage(stage: int, mode: GameMode, theme: String) -> void:
	current_stage = stage
	current_mode = mode
	current_theme = theme
	_play_start_time = Time.get_ticks_msec() / 1000.0
	_hint_count_used = 0
	_words_found_count = 0
	_is_combo = false

	_change_state(GameState.PLAYING)
	game_started.emit(stage, mode)


## 게임을 일시 정지한다.
func pause() -> void:
	if current_state != GameState.PLAYING:
		return
	_change_state(GameState.PAUSED)
	game_paused.emit()


## 게임을 재개한다.
func resume() -> void:
	if current_state != GameState.PAUSED:
		return
	_change_state(GameState.PLAYING)
	game_resumed.emit()


## 스테이지를 클리어 처리한다.
func complete_stage() -> void:
	if current_state != GameState.PLAYING:
		return

	var elapsed := (Time.get_ticks_msec() / 1000.0) - _play_start_time
	var result := GameResult.new()
	result.stage = current_stage
	result.mode = current_mode
	result.theme = current_theme
	result.clear_time = elapsed
	result.hints_used = _hint_count_used
	result.words_found = _words_found_count
	result.total_words = _total_words
	result.is_cleared = true

	_change_state(GameState.COMPLETED)
	game_completed.emit(result)


## Time Attack 시간 초과로 스테이지 실패 처리한다.
func fail_stage() -> void:
	if current_state != GameState.PLAYING:
		return

	var elapsed := (Time.get_ticks_msec() / 1000.0) - _play_start_time
	var result := GameResult.new()
	result.stage = current_stage
	result.mode = current_mode
	result.theme = current_theme
	result.clear_time = elapsed
	result.hints_used = _hint_count_used
	result.words_found = _words_found_count
	result.total_words = _total_words
	result.is_cleared = false

	_change_state(GameState.FAILED)
	game_failed.emit(result)


## 다음 스테이지로 진행한다.
func next_stage() -> void:
	if current_state != GameState.COMPLETED:
		return
	start_stage(current_stage + 1, current_mode, "")  # 테마는 ThemeRandomizer가 결정


## 현재 스테이지를 재시도한다.
func retry_stage() -> void:
	if current_state not in [GameState.COMPLETED, GameState.FAILED]:
		return
	start_stage(current_stage, current_mode, "")


## 홈(타이틀) 화면으로 돌아간다.
func go_home() -> void:
	if current_state == GameState.IDLE:
		return
	_change_state(GameState.IDLE)


## 단어 발견 시 호출한다.
func on_word_found() -> void:
	_words_found_count += 1


## 힌트 사용 시 호출한다.
func on_hint_used() -> void:
	_hint_count_used += 1


## 전체 단어 수를 설정한다.
func set_total_words(count: int) -> void:
	_total_words = count


## 앱 포커스 변경 시 자동 일시정지/재개 처리
func _notification(what: int) -> void:
	match what:
		NOTIFICATION_APPLICATION_FOCUS_OUT:
			if current_state == GameState.PLAYING:
				pause()
		NOTIFICATION_APPLICATION_FOCUS_IN:
			# 사용자가 수동으로 resume() 호출해야 하므로 자동 재개하지 않음
			pass
```

### 1.6 GameResult 데이터 클래스

```gdscript
## game_result.gd
## 스테이지 결과를 담는 데이터 클래스.
## 포팅 원본: GameResult.cs
class_name GameResult
extends RefCounted

## 스테이지 번호
var stage: int = 0

## 게임 모드
var mode: int = 0  # GameControllerClass.GameMode

## 테마 이름
var theme: String = ""

## 클리어 소요 시간 (초)
var clear_time: float = 0.0

## 사용한 힌트 수
var hints_used: int = 0

## 발견한 단어 수
var words_found: int = 0

## 전체 단어 수
var total_words: int = 0

## 클리어 여부 (true: 성공, false: 실패)
var is_cleared: bool = false

## 획득 코인
var coins_earned: int = 0

## 신기록 여부
var is_new_record: bool = false

## 클리어 등급 (S/A/B/C)
var grade: String = "C"


## 등급을 계산하여 설정한다.
func calculate_grade() -> void:
	if hints_used == 0 and clear_time < 90.0:
		grade = "S"
	elif hints_used == 0:
		grade = "A"
	elif hints_used == 1:
		grade = "B"
	else:
		grade = "C"


## 등급 보상 배율을 반환한다.
func get_grade_multiplier() -> float:
	match grade:
		"S": return 2.0
		"A": return 1.5
		"B": return 1.0
		"C": return 0.8
		_: return 1.0


## 기본 코인 보상을 계산한다.
func calculate_coins() -> int:
	if not is_cleared:
		return 0
	var base_coins: int = 10 + (stage * 2)
	coins_earned = int(base_coins * get_grade_multiplier())
	return coins_earned


## Dictionary로 직렬화한다 (저장용).
func to_dict() -> Dictionary:
	return {
		"stage": stage,
		"mode": mode,
		"theme": theme,
		"clear_time": clear_time,
		"hints_used": hints_used,
		"words_found": words_found,
		"total_words": total_words,
		"is_cleared": is_cleared,
		"coins_earned": coins_earned,
		"is_new_record": is_new_record,
		"grade": grade,
	}
```

### 2.2 화면 전환 유형 설명

```
Root
├── TabLayer (5개 탭 화면)
│   ├── DailyChallengeScreen (탭 0) — Lv.24 해금
│   ├── TeamScreen (탭 1) — Lv.44 해금 (잠금 화면)
│   ├── TitleScreen/Home (탭 2, 기본)
│   ├── CollectionScreen (탭 3) — Coming Soon
│   └── ShopScreen (탭 4)
├── PushLayer
│   ├── GameScreen
│   └── ResultScreen
├── PopupLayer
│   ├── SettingsPopup
│   ├── ProfilePopup (신규 — Stats 포함)
│   └── AvatarSelectPopup
└── BottomTabBar (5탭)
```

**Tab 화면:**
- 항상 메모리에 유지되며 `show()` / `hide()`로 전환한다.
- Tab 화면 간 전환 시 기존 상태를 보존한다.
- Bottom Tab Bar에 의해 전환된다.

**Push 화면:**
- 스택 기반으로 위에 쌓인다.
- 뒤로 가기(Back)로 이전 화면 복귀가 가능하다.
- Push 화면이 활성화되면 아래 Tab 화면은 가려진다.

**Popup 화면:**
- 최상위 모달 오버레이로 표시된다.
- 반투명 배경(dim) 처리된다.
- 바깥 영역 탭 또는 닫기 버튼으로 해제한다.

## 3. ScreenManager 설계

### 3.1 base_screen.gd - 추상 기본 클래스

모든 화면은 `BaseScreen`을 상속하여 `enter()` / `exit()` 생명주기를 구현한다.

```gdscript
## base_screen.gd
## 모든 화면(Tab, Push, Popup)이 상속하는 추상 기본 클래스.
## enter()/exit() 생명주기를 통해 화면 전환 시 초기화/정리를 수행한다.
class_name BaseScreen
extends Control


## 화면이 활성화될 때 호출된다.
## 전환 애니메이션 시작 전에 호출되므로 UI 초기화를 수행한다.
## data: 이전 화면에서 전달하는 임의 데이터 (없으면 빈 Dictionary)
func enter(data: Dictionary = {}) -> void:
	pass


## 화면이 비활성화될 때 호출된다.
## 전환 애니메이션 종료 후에 호출된다.
func exit() -> void:
	pass


## 뒤로 가기(Back) 입력을 처리한다.
## true를 반환하면 ScreenManager가 기본 동작(pop)을 수행하지 않는다.
func handle_back() -> bool:
	return false
```

### 3.2 screen_manager.gd - 화면 전환 관리자

```gdscript
## screen_manager.gd (Autoload)
## 화면 전환을 관리하는 싱글톤.
## Tab 전환, Push 네비게이션, Popup 표시를 통합 관리한다.
## 포팅 원본: ScreenManager (TSD 3.2.11)
extends Node

## --- 시그널 ---
signal screen_changed(screen_name: String)
signal push_stack_changed(stack_size: int)
signal popup_opened(popup_name: String)
signal popup_closed(popup_name: String)

## --- 씬 프리로드 ---
## Tab 화면은 앱 시작 시 미리 로드하여 메모리에 유지한다.
var _tab_scenes: Dictionary = {
	"daily": preload("res://scenes/screens/daily_challenge_screen.tscn"),
	"team": preload("res://scenes/screens/team_screen.tscn"),
	"title": preload("res://scenes/screens/title_screen.tscn"),
	"collection": preload("res://scenes/screens/collection_screen.tscn"),
	"shop": preload("res://scenes/screens/shop_screen.tscn"),
}

## Push 화면은 필요 시 로드한다 (미리 프리로드 가능).
var _push_scenes: Dictionary = {
	"game": preload("res://scenes/screens/game_screen.tscn"),
	"result": preload("res://scenes/screens/result_screen.tscn"),
}

## Popup 씬
var _popup_scenes: Dictionary = {
	"settings": preload("res://scenes/popups/settings_popup.tscn"),
	"profile": preload("res://scenes/popups/profile_popup.tscn"),
	"avatar_select": preload("res://scenes/popups/avatar_select_popup.tscn"),
}

## --- 레이어 노드 참조 ---
## main_scene.tscn의 레이어 노드를 _ready()에서 참조한다.
var _tab_layer: Control = null
var _push_layer: Control = null
var _popup_layer: Control = null
var _bottom_tab_bar: Control = null

## --- 상태 ---
var _tab_instances: Dictionary = {}      # {String: BaseScreen}
var _active_tab: String = ""
var _push_stack: Array = []              # Array[BaseScreen]
var _active_popup: BaseScreen = null

## --- 전환 애니메이션 설정 ---
const TRANSITION_DURATION: float = 0.3
const FADE_DURATION: float = 0.2


func _ready() -> void:
	# main_scene.tscn에서 레이어 노드를 찾는다.
	# main_scene의 구조:
	#   MainScene (Control)
	#     ├── TabLayer (Control)
	#     ├── PushLayer (Control)
	#     ├── PopupLayer (Control)
	#     └── BottomTabBar (HBoxContainer)
	await get_tree().process_frame  # 씬 트리가 준비될 때까지 대기
	var main := get_tree().current_scene
	_tab_layer = main.get_node("TabLayer")
	_push_layer = main.get_node("PushLayer")
	_popup_layer = main.get_node("PopupLayer")
	_bottom_tab_bar = main.get_node("BottomTabBar")

	# Tab 화면 인스턴스 생성 (최초 1회)
	_initialize_tabs()

	# 기본 Tab 표시
	switch_tab("title")


## Tab 화면 인스턴스를 모두 생성하고 숨겨둔다.
func _initialize_tabs() -> void:
	for tab_name in _tab_scenes:
		var instance: BaseScreen = _tab_scenes[tab_name].instantiate()
		instance.visible = false
		_tab_layer.add_child(instance)
		_tab_instances[tab_name] = instance


## --- Tab 전환 ---

## Tab 화면을 전환한다. 기존 Tab은 숨기고 새 Tab을 표시한다.
func switch_tab(tab_name: String) -> void:
	if tab_name == _active_tab:
		return
	if tab_name not in _tab_instances:
		push_warning("ScreenManager: 알 수 없는 Tab '%s'" % tab_name)
		return

	# Push 스택이 있으면 모두 제거
	_clear_push_stack()

	# 기존 Tab 숨기기
	if _active_tab != "" and _active_tab in _tab_instances:
		var old_tab: BaseScreen = _tab_instances[_active_tab]
		old_tab.exit()
		_animate_fade_out(old_tab)

	# 새 Tab 표시
	_active_tab = tab_name
	var new_tab: BaseScreen = _tab_instances[tab_name]
	new_tab.visible = true
	new_tab.enter()
	_animate_fade_in(new_tab)

	# Bottom Tab Bar 표시
	_bottom_tab_bar.visible = true
	_update_tab_bar_highlight(tab_name)

	screen_changed.emit(tab_name)


## --- Push 네비게이션 ---

## Push 화면을 스택에 추가하고 표시한다.
func push_screen(screen_name: String, data: Dictionary = {}) -> void:
	if screen_name not in _push_scenes:
		push_warning("ScreenManager: 알 수 없는 Push 화면 '%s'" % screen_name)
		return

	var instance: BaseScreen = _push_scenes[screen_name].instantiate()
	_push_layer.add_child(instance)
	_push_stack.append(instance)

	# 화면 초기화
	instance.enter(data)

	# 슬라이드 인 애니메이션
	_animate_slide_in(instance)

	# Bottom Tab Bar 숨기기
	_bottom_tab_bar.visible = false

	push_stack_changed.emit(_push_stack.size())
	screen_changed.emit(screen_name)


## Push 스택에서 최상위 화면을 제거하고 이전 화면으로 복귀한다.
func pop_screen() -> void:
	if _push_stack.is_empty():
		return

	var top_screen: BaseScreen = _push_stack.pop_back()
	top_screen.exit()

	# 슬라이드 아웃 애니메이션 후 제거
	_animate_slide_out(top_screen, func() -> void:
		top_screen.queue_free()
	)

	# 스택이 비면 Tab 화면 + Bottom Tab Bar 복원
	if _push_stack.is_empty():
		_bottom_tab_bar.visible = true
		if _active_tab in _tab_instances:
			_tab_instances[_active_tab].enter()

	push_stack_changed.emit(_push_stack.size())


## Push 스택을 모두 비운다.
func _clear_push_stack() -> void:
	while not _push_stack.is_empty():
		var screen: BaseScreen = _push_stack.pop_back()
		screen.exit()
		screen.queue_free()
	push_stack_changed.emit(0)


## --- Popup ---

## Popup을 표시한다.
func open_popup(popup_name: String, data: Dictionary = {}) -> void:
	if popup_name not in _popup_scenes:
		push_warning("ScreenManager: 알 수 없는 Popup '%s'" % popup_name)
		return

	# 기존 Popup이 열려있으면 닫기
	if _active_popup != null:
		close_popup()

	var instance: BaseScreen = _popup_scenes[popup_name].instantiate()
	_popup_layer.add_child(instance)
	_active_popup = instance

	instance.enter(data)
	_animate_popup_in(instance)

	popup_opened.emit(popup_name)


## 현재 열린 Popup을 닫는다.
func close_popup() -> void:
	if _active_popup == null:
		return

	var popup := _active_popup
	_active_popup = null
	popup.exit()

	_animate_popup_out(popup, func() -> void:
		popup.queue_free()
	)

	popup_closed.emit("")


## --- 전환 애니메이션 ---

## 페이드 인 애니메이션
func _animate_fade_in(target: Control) -> void:
	target.modulate.a = 0.0
	target.visible = true
	var tw := create_tween()
	tw.tween_property(target, "modulate:a", 1.0, FADE_DURATION)


## 페이드 아웃 애니메이션
func _animate_fade_out(target: Control) -> void:
	var tw := create_tween()
	tw.tween_property(target, "modulate:a", 0.0, FADE_DURATION)
	tw.tween_callback(func() -> void:
		target.visible = false
		target.modulate.a = 1.0  # 다음 표시를 위해 복원
	)


## Push 화면 슬라이드 인 (오른쪽에서 왼쪽)
func _animate_slide_in(target: Control) -> void:
	var screen_width := get_viewport().get_visible_rect().size.x
	target.position.x = screen_width
	target.modulate.a = 0.0
	var tw := create_tween().set_parallel(true)
	tw.tween_property(target, "position:x", 0.0, TRANSITION_DURATION)\
		.set_ease(Tween.EASE_OUT).set_trans(Tween.TRANS_CUBIC)
	tw.tween_property(target, "modulate:a", 1.0, FADE_DURATION)


## Push 화면 슬라이드 아웃 (왼쪽에서 오른쪽)
func _animate_slide_out(target: Control, on_complete: Callable) -> void:
	var screen_width := get_viewport().get_visible_rect().size.x
	var tw := create_tween().set_parallel(true)
	tw.tween_property(target, "position:x", screen_width, TRANSITION_DURATION)\
		.set_ease(Tween.EASE_IN).set_trans(Tween.TRANS_CUBIC)
	tw.tween_property(target, "modulate:a", 0.0, FADE_DURATION)
	tw.chain().tween_callback(on_complete)


## Popup 페이드 + 스케일 인
func _animate_popup_in(target: Control) -> void:
	target.modulate.a = 0.0
	target.scale = Vector2(0.9, 0.9)
	target.pivot_offset = target.size / 2.0
	var tw := create_tween().set_parallel(true)
	tw.tween_property(target, "modulate:a", 1.0, FADE_DURATION)
	tw.tween_property(target, "scale", Vector2.ONE, TRANSITION_DURATION)\
		.set_ease(Tween.EASE_OUT).set_trans(Tween.TRANS_BACK)


## Popup 페이드 + 스케일 아웃
func _animate_popup_out(target: Control, on_complete: Callable) -> void:
	var tw := create_tween().set_parallel(true)
	tw.tween_property(target, "modulate:a", 0.0, FADE_DURATION)
	tw.tween_property(target, "scale", Vector2(0.9, 0.9), TRANSITION_DURATION)\
		.set_ease(Tween.EASE_IN).set_trans(Tween.TRANS_CUBIC)
	tw.chain().tween_callback(on_complete)


## Tab Bar 하이라이트 갱신
func _update_tab_bar_highlight(tab_name: String) -> void:
	if _bottom_tab_bar and _bottom_tab_bar.has_method("set_active_tab"):
		_bottom_tab_bar.set_active_tab(tab_name)


## --- 입력 처리 ---

## Android 뒤로 가기 버튼 처리
func _unhandled_input(event: InputEvent) -> void:
	if event.is_action_pressed("ui_cancel"):
		# Popup이 열려있으면 Popup 닫기
		if _active_popup != null:
			close_popup()
			get_viewport().set_input_as_handled()
			return

		# Push 화면이 있으면 Pop
		if not _push_stack.is_empty():
			var top: BaseScreen = _push_stack.back()
			if not top.handle_back():
				pop_screen()
			get_viewport().set_input_as_handled()
			return
```

```gdscript
## bottom_tab_bar.gd
## 하단 네비게이션 탭 바.
## TSD 3.2.23 BottomTabBar 기반.
extends HBoxContainer

## --- 시그널 ---
signal tab_selected(tab_name: String)

## --- 탭 매핑 ---
const TAB_CONFIG: Array[Dictionary] = [
	{"name": "daily", "label": "Daily", "icon": "calendar", "enabled": true, "unlock_level": 24},
	{"name": "team", "label": "Team", "icon": "shield", "enabled": true, "unlock_level": 44},
	{"name": "title", "label": "Home", "icon": "home", "enabled": true, "unlock_level": 0},
	{"name": "collection", "label": "Collection", "icon": "suitcase", "enabled": false, "unlock_level": -1},
	{"name": "shop", "label": "Shop", "icon": "cart", "enabled": true, "unlock_level": 0},
]

## --- 노드 참조 ---
## 각 탭 버튼은 씬에서 배치한다.
@onready var tab_buttons: Array[Button] = [
	$DailyTab, $TeamTab, $HomeTab, $CollectionTab, $ShopTab
]

## --- 상태 ---
var _active_tab_name: String = "title"

## --- 스트릭 Badge ---
@onready var daily_badge: Label = $DailyTab/Badge


func _ready() -> void:
	for i in range(tab_buttons.size()):
		var config := TAB_CONFIG[i]
		var button := tab_buttons[i]
		button.disabled = not config.enabled
		button.pressed.connect(_on_tab_pressed.bind(config.name))

	set_active_tab("title")


## 활성 탭을 설정한다. 해당 탭 버튼을 하이라이트 처리한다.
func set_active_tab(tab_name: String) -> void:
	_active_tab_name = tab_name
	for i in range(tab_buttons.size()):
		var button := tab_buttons[i]
		var config := TAB_CONFIG[i]
		if config.name == tab_name:
			button.add_theme_color_override("font_color", Color("#FF6B35"))
			# 선택 인디케이터 표시
		else:
			button.remove_theme_color_override("font_color")


## Daily 탭의 스트릭 Badge를 업데이트한다.
func update_daily_badge(streak: int) -> void:
	if streak > 0:
		daily_badge.text = str(streak)
		daily_badge.visible = true
	else:
		daily_badge.visible = false


## 탭 버튼 클릭 핸들러
func _on_tab_pressed(tab_name: String) -> void:
	if tab_name == _active_tab_name:
		return
	# 비활성 탭은 무시
	for config in TAB_CONFIG:
		if config.name == tab_name and not config.enabled:
			return

	set_active_tab(tab_name)
	tab_selected.emit(tab_name)

	# ScreenManager에 탭 전환 요청
	ScreenManager.switch_tab(tab_name)
```

## 5. PopupManager

Popup은 ScreenManager가 통합 관리하되, 공통 기능(배경 딤, 외부 탭 닫기)을 PopupManager가 담당한다.

### 5.1 popup_overlay.gd

```gdscript
## popup_overlay.gd
## 모든 Popup의 부모 컨테이너.
## 반투명 배경(dim), 바깥 탭 시 닫기 기능을 제공한다.
extends Control

## --- 상수 ---
const DIM_COLOR: Color = Color(0, 0, 0, 0.5)
const DIM_FADE_DURATION: float = 0.2

## --- 노드 ---
@onready var dim_background: ColorRect = $DimBackground
@onready var content_container: CenterContainer = $ContentContainer


func _ready() -> void:
	# 전체 화면 크기로 설정
	anchors_preset = Control.PRESET_FULL_RECT

	# 딤 배경 초기화
	dim_background.color = Color(0, 0, 0, 0)
	dim_background.anchors_preset = Control.PRESET_FULL_RECT
	dim_background.gui_input.connect(_on_dim_background_input)


## Popup을 표시한다.
func show_popup(popup_content: Control) -> void:
	content_container.add_child(popup_content)
	visible = true

	# 딤 배경 페이드 인
	var tw := create_tween()
	tw.tween_property(dim_background, "color:a", DIM_COLOR.a, DIM_FADE_DURATION)


## Popup을 닫는다.
func hide_popup() -> void:
	var tw := create_tween()
	tw.tween_property(dim_background, "color:a", 0.0, DIM_FADE_DURATION)
	tw.tween_callback(func() -> void:
		# 콘텐츠 제거
		for child in content_container.get_children():
			child.queue_free()
		visible = false
	)


## 딤 배경 클릭 시 Popup 닫기
func _on_dim_background_input(event: InputEvent) -> void:
	if event is InputEventMouseButton:
		var mouse_event := event as InputEventMouseButton
		if mouse_event.pressed and mouse_event.button_index == MOUSE_BUTTON_LEFT:
			ScreenManager.close_popup()
```
