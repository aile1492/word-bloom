## 1. 로컬 저장 구조

### 1.1 저장 경로

Godot의 `user://` 디렉토리에 JSON 파일로 저장한다.

| 항목 | 값 |
|------|----|
| **저장 파일** | `user://save_data.json` |
| **백업 파일** | `user://save_data.backup.json` |
| **체크섬 파일** | `user://save_data.sha256` (선택) |

**플랫폼별 실제 경로:**

| 플랫폼 | `user://` 실제 경로 |
|--------|-------------------|
| Android | `/data/data/com.yourapp.wordsearch/files/` |
| iOS | `Documents/` (앱 샌드박스 내) |
| Windows (개발) | `%APPDATA%/Godot/app_userdata/WordSearchPuzzle/` |

### 1.2 파일 형식

JSON 형식을 사용한다. 사람이 읽을 수 있고, 디버깅이 용이하며, Godot의 `JSON` 클래스로 직접 파싱 가능하다.

```
user://
├── save_data.json           ← 메인 세이브 파일
├── save_data.backup.json    ← 이전 세이브 백업
└── save_data.sha256         ← SHA-256 체크섬 (선택)
```

### 1.3 Godot FileAccess 기본 사용법

```gdscript
## 파일 쓰기
func _write_json(path: String, data: Dictionary) -> bool:
	var json_string: String = JSON.stringify(data, "\t")
	var file: FileAccess = FileAccess.open(path, FileAccess.WRITE)
	if file == null:
		push_error("SaveManager: 파일 열기 실패 - %s (에러: %s)" % [
			path, error_string(FileAccess.get_open_error())
		])
		return false
	file.store_string(json_string)
	file.close()
	return true


## 파일 읽기
func _read_json(path: String) -> Dictionary:
	if not FileAccess.file_exists(path):
		return {}
	var file: FileAccess = FileAccess.open(path, FileAccess.READ)
	if file == null:
		push_error("SaveManager: 파일 읽기 실패 - %s" % path)
		return {}
	var json_string: String = file.get_as_text()
	file.close()

	var json := JSON.new()
	var parse_result: Error = json.parse(json_string)
	if parse_result != OK:
		push_error("SaveManager: JSON 파싱 실패 - line %d: %s" % [
			json.get_error_line(), json.get_error_message()
		])
		return {}

	if json.data is Dictionary:
		return json.data
	return {}
```

### 3.2 load() - 데이터 불러오기

```gdscript
## 저장 파일을 읽어 메모리에 로드한다.
## 파일이 없으면 기본 데이터를 생성한다.
## 파일이 손상되었으면 백업에서 복원을 시도한다.
func load() -> void:
	# 1. 메인 파일 시도
	_data = _read_json(SAVE_PATH)

	# 2. 메인 파일 실패 시 백업 시도
	if _data.is_empty():
		push_warning("SaveManager: 메인 파일 로드 실패, 백업 시도")
		_data = _read_json(BACKUP_PATH)

		if not _data.is_empty():
			# 백업에서 복원 성공 → 메인 파일로 복사
			_write_json(SAVE_PATH, _data)
			print("SaveManager: 백업에서 복원 성공")

	# 3. 백업도 실패 시 기본 데이터 생성
	if _data.is_empty():
		print("SaveManager: 저장 파일 없음, 기본 데이터 생성")
		_data = _get_default_data()
		save()

	# 4. 마이그레이션 확인
	var file_version: int = _data.get("version", 0)
	if file_version < CURRENT_VERSION:
		_migrate_data(file_version)

	# 5. 누락 필드 보정 (새 필드가 추가된 경우)
	_ensure_default_fields()

	_is_loaded = true
	_is_dirty = false
	data_loaded.emit()

	if OS.is_debug_build():
		print("SaveManager: 데이터 로드 완료 (version: %d, stage: %d)" % [
			_data.get("version", 0),
			_data.get("current_stage", 1)
		])


## 기본 데이터 구조에 존재하는 필드가 로드된 데이터에 없으면 기본값으로 추가한다.
func _ensure_default_fields() -> void:
	var defaults := _get_default_data()
	_merge_defaults(_data, defaults)


## 재귀적으로 기본값을 병합한다.
## 기존 데이터에 없는 키만 추가하고, 기존 값은 유지한다.
func _merge_defaults(target: Dictionary, defaults: Dictionary) -> void:
	for key in defaults:
		if key not in target:
			target[key] = defaults[key]
		elif target[key] is Dictionary and defaults[key] is Dictionary:
			_merge_defaults(target[key], defaults[key])
```

### 3.3 save() - 데이터 저장

```gdscript
## 현재 메모리의 데이터를 파일로 저장한다.
## 저장 전 기존 파일을 백업한다.
func save() -> void:
	if not _is_loaded:
		push_warning("SaveManager: 로드되지 않은 상태에서 저장 시도")
		return

	# 1. 기존 파일을 백업으로 복사
	if FileAccess.file_exists(SAVE_PATH):
		var existing: FileAccess = FileAccess.open(SAVE_PATH, FileAccess.READ)
		if existing:
			var content: String = existing.get_as_text()
			existing.close()
			var backup: FileAccess = FileAccess.open(BACKUP_PATH, FileAccess.WRITE)
			if backup:
				backup.store_string(content)
				backup.close()

	# 2. 메인 파일 저장
	var success := _write_json(SAVE_PATH, _data)

	if success:
		# 3. 체크섬 저장 (선택)
		_save_checksum()

		_is_dirty = false
		data_saved.emit()

		if OS.is_debug_build():
			print("SaveManager: 저장 완료")
	else:
		push_error("SaveManager: 저장 실패!")
```

```gdscript
## --- Daily ---

func get_daily_streak() -> int:
	return _data.get("daily_streak", 0)

func update_daily_streak(completed_date: String) -> void:
	var last_date: String = _data.get("last_daily_date", "")

	if completed_date == last_date:
		# 이미 오늘 완료함
		return

	# 연속 여부 확인 (날짜 차이 1일이면 연속)
	if _is_consecutive_date(last_date, completed_date):
		_data["daily_streak"] = _data.get("daily_streak", 0) + 1
	else:
		_data["daily_streak"] = 1

	_data["last_daily_date"] = completed_date
	_is_dirty = true


## 두 날짜가 연속(1일 차이)인지 확인한다.
func _is_consecutive_date(prev_date: String, curr_date: String) -> bool:
	if prev_date.is_empty():
		return false
	# 간단한 날짜 비교 (ISO 8601 형식: "2026-03-11")
	# Godot에는 Date 파싱이 제한적이므로 Unix timestamp 변환 사용
	var prev_dict := Time.get_datetime_dict_from_datetime_string(prev_date + "T00:00:00", false)
	var curr_dict := Time.get_datetime_dict_from_datetime_string(curr_date + "T00:00:00", false)
	var prev_unix := Time.get_unix_time_from_datetime_dict(prev_dict)
	var curr_unix := Time.get_unix_time_from_datetime_dict(curr_dict)
	var diff_days := int((curr_unix - prev_unix) / 86400.0)
	return diff_days == 1

	return _data.get("daily_login", {}).duplicate(true)

func update_daily_login(login_data: Dictionary) -> void:
	_data["daily_login"] = login_data
	_is_dirty = true


## --- 데이터 초기화 ---

func reset_data() -> void:
	_data = _get_default_data()
	_is_dirty = true
	save()
	data_reset.emit()
	print("SaveManager: 데이터 초기화 완료")
```

### 3.5 자동 저장 (_notification)

```gdscript
## 앱 종료 또는 백그라운드 진입 시 미저장 데이터를 자동으로 저장한다.
func _notification(what: int) -> void:
	match what:
		NOTIFICATION_WM_CLOSE_REQUEST:
			if _is_dirty:
				save()
		NOTIFICATION_APPLICATION_FOCUS_OUT:
			if _is_dirty:
				save()
		NOTIFICATION_APPLICATION_PAUSE_REQUEST:
			if _is_dirty:
				save()


func _ready() -> void:
	# 앱 시작 시 자동 로드
	load()
```

## 5. 데이터 무결성

### 5.1 SHA-256 체크섬 (선택적)

저장 파일의 위변조를 감지하기 위해 SHA-256 체크섬을 사용한다. 이 기능은 선택적이며, 필요에 따라 활성화한다.

```gdscript
## 체크섬 저장
func _save_checksum() -> void:
	var json_string: String = JSON.stringify(_data)
	var hash: String = json_string.sha256_text()
	var file: FileAccess = FileAccess.open(CHECKSUM_PATH, FileAccess.WRITE)
	if file:
		file.store_string(hash)
		file.close()


## 체크섬 검증
func _verify_checksum() -> bool:
	if not FileAccess.file_exists(CHECKSUM_PATH):
		return true  # 체크섬 파일 없으면 검증 생략

	if not FileAccess.file_exists(SAVE_PATH):
		return false

	# 저장 파일의 해시 계산
	var save_file: FileAccess = FileAccess.open(SAVE_PATH, FileAccess.READ)
	if save_file == null:
		return false
	var save_content: String = save_file.get_as_text()
	save_file.close()

	# JSON 재파싱 후 다시 직렬화하여 비교 (포맷 차이 방지)
	var json := JSON.new()
	if json.parse(save_content) != OK:
		return false
	var normalized: String = JSON.stringify(json.data)
	var calculated_hash: String = normalized.sha256_text()

	# 저장된 해시 읽기
	var hash_file: FileAccess = FileAccess.open(CHECKSUM_PATH, FileAccess.READ)
	if hash_file == null:
		return false
	var stored_hash: String = hash_file.get_as_text().strip_edges()
	hash_file.close()

	return calculated_hash == stored_hash
```

### 5.2 손상 복구 전략

```
[로드 시도]
     │
     ├── 메인 파일 정상 → 로드 완료
     │
     ├── 메인 파일 손상/없음
     │       │
     │       ├── 백업 파일 정상 → 백업에서 복원
     │       │
     │       └── 백업도 손상/없음 → 기본 데이터 생성 (첫 실행 취급)
     │
     └── 체크섬 불일치
             │
             ├── 백업 파일 정상 → 백업에서 복원
             │
             └── 백업도 불일치 → 메인 파일 사용 + 경고 로그
```

### 5.3 데이터 범위 검증

로드 후 주요 필드의 값이 유효한 범위 내인지 검증한다.

```gdscript
## 로드된 데이터의 범위를 검증하고 수정한다.
func _validate_data_ranges() -> void:
	# 코인은 0 이상
	if _data.get("coin_balance", 0) < 0:
		_data["coin_balance"] = 0

	# 스테이지는 1 이상
	if _data.get("current_stage", 1) < 1:
		_data["current_stage"] = 1

	# 아바타 인덱스 범위
	var avatar_idx: int = _data.get("selected_avatar_index", 0)
	if avatar_idx < 0 or avatar_idx > 20:
		_data["selected_avatar_index"] = 0

	# 스트릭은 0 이상
	if _data.get("daily_streak", 0) < 0:
		_data["daily_streak"] = 0

	# best_times는 0 이상
	var best_times: Dictionary = _data.get("best_times", {})
	for key in best_times:
		if best_times[key] < 0.0:
			best_times[key] = 999.0
	_data["best_times"] = best_times

	# 설정값 범위
	var settings: Dictionary = _data.get("settings", {})
	settings["language_index"] = clampi(settings.get("language_index", 0), 0, 1)
	settings["font_size_index"] = clampi(settings.get("font_size_index", 1), 0, 2)
	_data["settings"] = settings
```

## 6. 데이터 마이그레이션

### 6.1 버전 관리 전략

`version` 필드를 사용하여 스키마 변경 시 자동 마이그레이션을 수행한다. 마이그레이션은 단계적으로 적용되어 어떤 버전에서든 최신 버전으로 업그레이드할 수 있다.

```gdscript
## 저장 데이터를 현재 버전으로 마이그레이션한다.
## 각 버전 단계를 순차적으로 적용한다.
func _migrate_data(from_version: int) -> void:
	print("SaveManager: 마이그레이션 시작 v%d → v%d" % [from_version, CURRENT_VERSION])

	var version := from_version

	# v0 → v1: 초기 스키마 (해당 없음, 기본 데이터 사용)
	if version < 1:
		_migrate_v0_to_v1()
		version = 1

	# 향후 마이그레이션 추가 예시:
	# if version < 2:
	#     _migrate_v1_to_v2()
	#     version = 2

	_data["version"] = CURRENT_VERSION
	_is_dirty = true

	print("SaveManager: 마이그레이션 완료 v%d" % CURRENT_VERSION)
```

### 6.2 마이그레이션 예시 (향후 v1 -> v2)

```gdscript
## v1 → v2 마이그레이션 예시
## 새 필드 추가: inventory 시스템
func _migrate_v1_to_v2() -> void:
	# 새 필드 추가
	if "inventory" not in _data:
		_data["inventory"] = {
			"hint_tokens": 0,
			"shuffle_tokens": 0,
			"time_freeze_tokens": 0,
		}

	# 기존 필드 이름 변경 예시
	if "sound_on" in _data.get("settings", {}):
		_data["settings"]["sound_enabled"] = _data["settings"]["sound_on"]
		_data["settings"].erase("sound_on")

	print("SaveManager: v1 → v2 마이그레이션 완료")
```

### 6.3 마이그레이션 안전 규칙

| 규칙 | 설명 |
|------|------|
| 기존 데이터 보존 | 마이그레이션은 기존 값을 절대 삭제하지 않는다 (이름 변경은 예외) |
| 순차 적용 | v1 → v3은 v1 → v2 → v3 순서로 적용 |
| 백업 후 진행 | 마이그레이션 전 자동 백업이 존재하므로 실패 시 복원 가능 |
| 테스트 필수 | 새 마이그레이션 추가 시 이전 버전 데이터로 테스트 |

## 7. 클라우드 동기화 (향후)

> **참고:** 클라우드 동기화는 v1.0 출시 후 업데이트에서 구현 예정이다. Godot Guide 섹션 13 (save/cloud) 기반 설계.

### 7.1 지원 플랫폼

| 플랫폼 | 서비스 | Godot 플러그인 |
|--------|--------|---------------|
| Android | Google Play Games Saved Games | GodotPlayGameServices 플러그인 |
| iOS | iCloud (CloudKit) | 커스텀 GDExtension 필요 |

### 7.2 동기화 전략

```
[앱 시작]
     │
     ├── 로컬 데이터 로드
     ├── 클라우드 데이터 조회
     │
     ├── 로컬만 존재 → 클라우드에 업로드
     ├── 클라우드만 존재 → 로컬에 다운로드
     ├── 둘 다 존재
     │     │
     │     ├── 동일 → 동기화 불필요
     │     │
     │     └── 다름 → 충돌 해결
     │           │
     │           ├── 자동: 더 높은 스테이지 / 더 많은 코인 쪽 선택
     │           └── 수동: 사용자에게 선택 팝업 표시
     │
     └── [플레이]
           │
           └── 스테이지 클리어 시 → 로컬 저장 + 클라우드 업로드 (비동기)
```

### 7.3 클라우드 데이터 형식

로컬 `save_data.json`과 동일한 JSON 구조를 사용한다. `last_sync_timestamp` 필드를 추가하여 동기화 시점을 추적한다.

```json
{
  "version": 1,
  "last_sync_timestamp": "2026-03-11T15:30:00Z",
  "...": "(save_data.json과 동일한 구조)"
}
```

### 7.4 인터페이스 설계 (향후 구현)

```gdscript
## cloud_sync_manager.gd (향후 구현)
## 클라우드 동기화 인터페이스.
## 플랫폼별 구현체를 분리한다.
class_name CloudSyncManager
extends Node

signal sync_started()
signal sync_completed(success: bool)
signal sync_conflict(local_data: Dictionary, cloud_data: Dictionary)

## 클라우드 데이터를 다운로드하여 로컬과 병합한다.
func sync() -> void:
	pass

## 로컬 데이터를 클라우드에 업로드한다.
func upload() -> void:
	pass

## 클라우드 데이터를 다운로드한다.
func download() -> Dictionary:
	return {}

## 충돌 시 더 진행된 데이터를 선택한다.
func resolve_conflict(local: Dictionary, cloud: Dictionary) -> Dictionary:
	var local_stage: int = local.get("current_stage", 1)
	var cloud_stage: int = cloud.get("current_stage", 1)
	# 더 높은 스테이지의 데이터를 선택
	if local_stage >= cloud_stage:
		return local
	return cloud
```
