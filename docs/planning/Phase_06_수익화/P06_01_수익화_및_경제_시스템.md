### 1.2 용어 정의

| 용어 | 설명 |
|------|------|
| **Coin** | 게임 내 단일 재화. 플레이, 광고, IAP를 통해 획득 |
| **IAP** | In-App Purchase, 인앱 구매 |
| **Rewarded Ad** | 보상형 광고. 유저 자발적 시청 후 보상 지급 |
| **Interstitial Ad** | 전면 광고. 자연스러운 전환 지점에서 노출 |
| **Premium Pass** | 월간/연간 구독 상품 |
| **F2P** | Free-to-Play, 무과금 유저 |
| **P2W** | Pay-to-Win, 과금이 승리에 직결되는 구조 (본 게임에서 지양) |

### 1.3 설계 원칙

1. **공정성 우선**: 과금은 편의 기능에만 적용, 게임 진행에 필수가 아님
2. **자연스러운 흐름**: 광고와 구매 유도가 게임플레이를 방해하지 않음
3. **투명한 가치**: 유저가 무엇을 얻는지 명확히 인지할 수 있도록 설계
4. **지속 가능성**: F2P 유저도 충분히 즐길 수 있으면서 과금 유인 존재

### 3.3 coin_manager.gd (GDScript Pseudocode)

```gdscript
## coin_manager.gd
## 코인 재화 관리 Autoload Singleton
## 코인 획득/소비/잔액 관리 및 이벤트 발행
extends Node

# ─── Signals ─────────────────────────────────────────────
signal coins_changed(new_amount: int, delta: int)
signal coins_insufficient(required: int, current: int)
signal transaction_completed(transaction: Dictionary)

# ─── Constants ───────────────────────────────────────────
const INITIAL_COINS: int = 300

## 스테이지 등급별 보상
const STAGE_REWARDS: Dictionary = {
	"S": 50,
	"A": 30,
	"B": 20,
	"C": 10,
}

## 일일 보상 (Day 1~7)
const DAILY_REWARDS: Array[int] = [30, 40, 50, 60, 80, 100, 200]

## 보상형 광고 보상
const REWARDED_AD_COINS: int = 50
const REWARDED_AD_DAILY_LIMIT: int = 15

## 보너스 워드 보상
const BONUS_WORD_REWARD: int = 20
const BONUS_WORD_COMPLETE_EXTRA: int = 10

## 힌트 비용
const HINT_COSTS: Dictionary = {
	"first_letter": 100,
	"full_reveal": 200,
	"magnifier": 150,
	"timer_extend": 80,
	"direction_show": 120,
}

## Premium Pass 할인율
const PREMIUM_DISCOUNT: float = 0.3

## 첫 3스테이지 일일 보너스 배율
const DAILY_FIRST_STAGE_MULTIPLIER: float = 1.5
const DAILY_FIRST_STAGE_COUNT: int = 3

## 연속 클리어 보너스
const CONSECUTIVE_CLEAR_THRESHOLD: int = 5
const CONSECUTIVE_CLEAR_BONUS: int = 20

# ─── State ───────────────────────────────────────────────
var _coins: int = 0
var _today_stages_cleared: int = 0
var _consecutive_clears: int = 0
var _today_rewarded_ads: int = 0
var _last_date_string: String = ""

# ─── Initialization ─────────────────────────────────────
func _ready() -> void:
	_load_from_save()
	_check_daily_reset()


func _load_from_save() -> void:
	var save_data: Dictionary = SaveManager.get_data()
	_coins = save_data.get("coins", INITIAL_COINS)
	_today_stages_cleared = save_data.get("today_stages_cleared", 0)
	_consecutive_clears = save_data.get("consecutive_clears", 0)
	_today_rewarded_ads = save_data.get("rewarded_ad_count_today", 0)
	_last_date_string = save_data.get("last_coin_date", "")


func _check_daily_reset() -> void:
	var today: String = Time.get_date_string_from_system()
	if today != _last_date_string:
		_today_stages_cleared = 0
		_today_rewarded_ads = 0
		_last_date_string = today
		_save_state()


# ─── Public API: 조회 ───────────────────────────────────
func get_coins() -> int:
	return _coins


func can_afford(amount: int) -> bool:
	return _coins >= amount


func get_hint_cost(hint_type: String) -> int:
	var base_cost: int = HINT_COSTS.get(hint_type, 0)
	if SubscriptionManager.is_premium_active():
		return int(base_cost * (1.0 - PREMIUM_DISCOUNT))
	return base_cost


# ─── Public API: 획득 ───────────────────────────────────
func add_stage_reward(grade: String) -> int:
	var base_reward: int = STAGE_REWARDS.get(grade, 0)
	var final_reward: int = base_reward

	# 일일 첫 3스테이지 보너스 적용
	_today_stages_cleared += 1
	if _today_stages_cleared <= DAILY_FIRST_STAGE_COUNT:
		final_reward = int(base_reward * DAILY_FIRST_STAGE_MULTIPLIER)

	# 연속 클리어 보너스 체크
	_consecutive_clears += 1
	if _consecutive_clears >= CONSECUTIVE_CLEAR_THRESHOLD:
		final_reward += CONSECUTIVE_CLEAR_BONUS
		_consecutive_clears = 0  # 리셋

	# Premium Pass 구독 시 일일 보상 2배는 daily_reward에만 적용
	_add_coins(final_reward, "stage_clear_%s" % grade)
	return final_reward


func add_daily_reward(day_index: int) -> int:
	## day_index: 0~6 (Day1~Day7)
	var clamped: int = clampi(day_index, 0, DAILY_REWARDS.size() - 1)
	var reward: int = DAILY_REWARDS[clamped]

	if SubscriptionManager.is_premium_active():
		reward *= 2

	_add_coins(reward, "daily_reward_day%d" % (clamped + 1))
	return reward


func add_achievement_reward(achievement_id: String, amount: int) -> int:
	_add_coins(amount, "achievement_%s" % achievement_id)
	return amount


func add_rewarded_ad_coins() -> int:
	_check_daily_reset()
	if _today_rewarded_ads >= REWARDED_AD_DAILY_LIMIT:
		push_warning("CoinManager: 일일 보상형 광고 한도 초과")
		return 0
	_today_rewarded_ads += 1
	_add_coins(REWARDED_AD_COINS, "rewarded_ad")
	return REWARDED_AD_COINS


func add_bonus_word_reward(all_found: bool) -> int:
	var reward: int = BONUS_WORD_REWARD
	if all_found:
		reward += BONUS_WORD_COMPLETE_EXTRA
	_add_coins(reward, "bonus_word")
	return reward


func add_coins_from_iap(amount: int) -> void:
	_add_coins(amount, "iap_purchase")


# ─── Public API: 소비 ───────────────────────────────────
func spend_for_streak_shield() -> bool:
	return _spend_coins(200, "streak_shield")


func spend_for_marathon_revive() -> bool:
	return _spend_coins(300, "marathon_revive")



func spend_for_hint(hint_type: String) -> bool:
	var cost: int = get_hint_cost(hint_type)
	return _spend_coins(cost, "hint_%s" % hint_type)


func spend_for_theme(theme_id: String, cost: int) -> bool:
	return _spend_coins(cost, "theme_%s" % theme_id)


func spend_for_customization(item_id: String, cost: int) -> bool:
	return _spend_coins(cost, "customization_%s" % item_id)


# ─── Internal ────────────────────────────────────────────
func _add_coins(amount: int, source: String) -> void:
	var old_amount: int = _coins
	_coins += amount
	_save_state()
	coins_changed.emit(_coins, amount)
	_log_transaction(source, amount, old_amount, _coins)


func _spend_coins(amount: int, purpose: String) -> bool:
	if not can_afford(amount):
		coins_insufficient.emit(amount, _coins)
		return false
	var old_amount: int = _coins
	_coins -= amount
	_save_state()
	coins_changed.emit(_coins, -amount)
	_log_transaction(purpose, -amount, old_amount, _coins)
	return true


func _save_state() -> void:
	SaveManager.set_fields({
		"coins": _coins,
		"today_stages_cleared": _today_stages_cleared,
		"consecutive_clears": _consecutive_clears,
		"rewarded_ad_count_today": _today_rewarded_ads,
		"last_coin_date": _last_date_string,
	})


func _log_transaction(source: String, delta: int, before: int, after: int) -> void:
	var transaction: Dictionary = {
		"source": source,
		"delta": delta,
		"before": before,
		"after": after,
		"timestamp": Time.get_unix_time_from_system(),
	}
	transaction_completed.emit(transaction)
	# 향후 Analytics 연동 지점
	print("[CoinManager] %s: %+d (잔액: %d)" % [source, delta, after])
```

## 5. 광고 전략

### 5.1 광고 유형 개요

| 유형 | 트리거 | 사용자 선택 | 빈도 제한 | 수익 기여 |
|------|--------|-----------|----------|----------|
| **Rewarded Ad** | 유저 자발적 클릭 | O (선택) | 15회/일 | 높음 |
| **Interstitial Ad** | 3스테이지마다 | X (자동) | 5회/세션 | 중간 |
| **Banner Ad** | 상시 (비게임 화면) | X (자동) | 항상 | 낮음 |

### 5.2 보상형 광고 (Rewarded Ad) 상세

#### 트리거 포인트

| 위치 | 트리거 | 보상 | UI 표현 |
|------|--------|------|---------|
| 결과 화면 | "보상 2배" 버튼 | 스테이지 보상 코인 2배 | 재생 아이콘 + "광고 시청하고 2배 받기" |
| 코인 상점 | "무료 코인" 버튼 | 50코인 | "광고 보고 50코인 받기" |
| 힌트 부족 시 | "무료 힌트" 버튼 | 힌트 1회 무료 사용 | "광고 보고 무료 힌트" |
| 마라톤 모드 실패 | "부활" 버튼 | 1회 부활 (코인 대체) | "광고 보고 이어하기" |
| 타이머 종료 직전 | "+30초" 버튼 | 시간 30초 추가 | "광고 보고 시간 추가" |

#### 제한 규칙

```
일일 시청 한도: 15회
├── 코인 보상 광고: 최대 10회
├── 힌트 무료 광고: 최대 5회
├── 마라톤 부활 광고: 최대 3회
└── 타이머 연장 광고: 최대 5회
    (각 카테고리 합산 15회 이내)

쿨다운:
├── 같은 유형: 30초
└── 다른 유형: 제한 없음
```

### 5.6 ad_manager.gd (GDScript Pseudocode)

```gdscript
## ad_manager.gd
## 광고 관리 Autoload Singleton
## 보상형 / 전면 / 배너 광고 통합 관리
extends Node

# ─── Signals ─────────────────────────────────────────────
signal rewarded_ad_completed(reward_type: String)
signal rewarded_ad_failed(reason: String)
signal interstitial_closed()
signal ad_revenue_earned(ad_type: String, value: float)

# ─── Enums ───────────────────────────────────────────────
enum AdType {
	REWARDED,
	INTERSTITIAL,
	BANNER,
}

enum RewardType {
	DOUBLE_COINS,
	FREE_COINS,
	FREE_HINT,
	MARATHON_REVIVE,
	TIMER_EXTEND,
}

# ─── Constants ───────────────────────────────────────────
const REWARDED_DAILY_LIMIT: int = 15
const REWARDED_COOLDOWN_SAME_TYPE_SEC: float = 30.0

const INTERSTITIAL_STAGE_INTERVAL: int = 3
const INTERSTITIAL_SESSION_LIMIT: int = 5
const INTERSTITIAL_MIN_INTERVAL_SEC: float = 120.0
const INTERSTITIAL_CLOSE_DELAY_SEC: float = 5.0

## 보상형 광고 카테고리별 하위 한도
const REWARDED_SUB_LIMITS: Dictionary = {
	"coin_reward": 10,
	"free_hint": 5,
	"marathon_revive": 3,
	"timer_extend": 5,
}

# ─── State ───────────────────────────────────────────────
var _ad_removal_purchased: bool = false
var _is_premium_subscriber: bool = false

## Rewarded Ad 상태
var _rewarded_count_today: int = 0
var _rewarded_sub_counts: Dictionary = {}  ## {"coin_reward": 2, "free_hint": 1, ...}
var _rewarded_last_shown: Dictionary = {}  ## {reward_type: timestamp}

## Interstitial 상태
var _interstitial_session_count: int = 0
var _interstitial_last_shown_time: float = 0.0
var _stages_since_last_interstitial: int = 0

## Banner 상태
var _banner_visible: bool = false

## Platform
var _is_tv_platform: bool = false
var _ad_plugin: Object = null  # 플랫폼별 광고 SDK 플러그인

# ─── Initialization ─────────────────────────────────────
func _ready() -> void:
	_load_state()
	_detect_platform()
	_initialize_ad_sdk()
	_check_daily_reset()


func _load_state() -> void:
	var save_data: Dictionary = SaveManager.get_data()
	_ad_removal_purchased = save_data.get("ad_removal", false)
	_is_premium_subscriber = SubscriptionManager.is_premium_active()
	_rewarded_count_today = save_data.get("rewarded_ad_count_today", 0)
	_rewarded_sub_counts = save_data.get("rewarded_sub_counts", {})
	var last_date: String = save_data.get("last_ad_date", "")
	if last_date != Time.get_date_string_from_system():
		_reset_daily_counts()


func _detect_platform() -> void:
	var os_name: String = OS.get_name()
	# Android TV, Fire TV 등 감지
	_is_tv_platform = os_name == "Android" and _check_tv_feature()


func _check_tv_feature() -> bool:
	# Android TV feature 확인 (android.software.leanback)
	if Engine.has_singleton("GodotAdMob"):
		return Engine.get_singleton("GodotAdMob").call("isTvDevice")
	return false


func _initialize_ad_sdk() -> void:
	if Engine.has_singleton("GodotAdMob"):
		_ad_plugin = Engine.get_singleton("GodotAdMob")
		_ad_plugin.connect("rewarded_ad_loaded", _on_rewarded_ad_loaded)
		_ad_plugin.connect("rewarded_ad_earned_reward", _on_rewarded_ad_earned)
		_ad_plugin.connect("rewarded_ad_failed", _on_rewarded_ad_failed)
		_ad_plugin.connect("interstitial_closed", _on_interstitial_closed)
		_ad_plugin.call("initialize")
		# Preload 광고
		_preload_rewarded_ad()
		_preload_interstitial()


func _reset_daily_counts() -> void:
	_rewarded_count_today = 0
	_rewarded_sub_counts.clear()
	for key in REWARDED_SUB_LIMITS:
		_rewarded_sub_counts[key] = 0
	_save_ad_state()


func _check_daily_reset() -> void:
	var today: String = Time.get_date_string_from_system()
	var last_date: String = SaveManager.get_data().get("last_ad_date", "")
	if today != last_date:
		_reset_daily_counts()
		SaveManager.set_fields({"last_ad_date": today})


# ─── Public API: 상태 확인 ──────────────────────────────
func is_ad_free() -> bool:
	return _ad_removal_purchased or _is_premium_subscriber


func can_show_rewarded(reward_category: String) -> bool:
	if _rewarded_count_today >= REWARDED_DAILY_LIMIT:
		return false
	var sub_count: int = _rewarded_sub_counts.get(reward_category, 0)
	var sub_limit: int = REWARDED_SUB_LIMITS.get(reward_category, 0)
	if sub_count >= sub_limit:
		return false
	# 쿨다운 체크
	var last_time: float = _rewarded_last_shown.get(reward_category, 0.0)
	var now: float = Time.get_unix_time_from_system()
	if now - last_time < REWARDED_COOLDOWN_SAME_TYPE_SEC:
		return false
	return true


func get_rewarded_remaining_today() -> int:
	return maxi(0, REWARDED_DAILY_LIMIT - _rewarded_count_today)


func should_show_interstitial() -> bool:
	if is_ad_free():
		return false
	if _is_tv_platform:
		# TV는 5스테이지마다
		return _stages_since_last_interstitial >= 5
	if _interstitial_session_count >= INTERSTITIAL_SESSION_LIMIT:
		return false
	if _stages_since_last_interstitial < INTERSTITIAL_STAGE_INTERVAL:
		return false
	var now: float = Time.get_unix_time_from_system()
	if now - _interstitial_last_shown_time < INTERSTITIAL_MIN_INTERVAL_SEC:
		return false
	return true


# ─── Public API: 광고 표시 ──────────────────────────────
func show_rewarded_ad(reward_category: String) -> void:
	if not can_show_rewarded(reward_category):
		rewarded_ad_failed.emit("limit_reached")
		return
	if _ad_plugin == null:
		# 디버그/에디터 환경: 즉시 보상
		_process_reward(reward_category)
		return
	_ad_plugin.call("showRewardedAd")
	# _pending_reward_category에 저장하여 콜백에서 처리
	_pending_reward_category = reward_category


func show_interstitial() -> void:
	if is_ad_free():
		interstitial_closed.emit()
		return
	if not should_show_interstitial():
		interstitial_closed.emit()
		return
	if _ad_plugin == null:
		interstitial_closed.emit()
		return
	_ad_plugin.call("showInterstitial")
	_interstitial_session_count += 1
	_interstitial_last_shown_time = Time.get_unix_time_from_system()
	_stages_since_last_interstitial = 0


func show_banner() -> void:
	if is_ad_free() or _is_tv_platform:
		return
	if _ad_plugin and not _banner_visible:
		_ad_plugin.call("showBanner")
		_banner_visible = true


func hide_banner() -> void:
	if _ad_plugin and _banner_visible:
		_ad_plugin.call("hideBanner")
		_banner_visible = false


func notify_stage_cleared() -> void:
	_stages_since_last_interstitial += 1


# ─── Public API: 구매 반영 ──────────────────────────────
func set_ad_removal(purchased: bool) -> void:
	_ad_removal_purchased = purchased
	if purchased:
		hide_banner()
	SaveManager.set_fields({"ad_removal": purchased})


func set_premium_status(active: bool) -> void:
	_is_premium_subscriber = active
	if active:
		hide_banner()


# ─── Internal: 보상 처리 ────────────────────────────────
var _pending_reward_category: String = ""


func _process_reward(reward_category: String) -> void:
	_rewarded_count_today += 1
	_rewarded_sub_counts[reward_category] = \
		_rewarded_sub_counts.get(reward_category, 0) + 1
	_rewarded_last_shown[reward_category] = Time.get_unix_time_from_system()
	_save_ad_state()
	rewarded_ad_completed.emit(reward_category)


func _save_ad_state() -> void:
	SaveManager.set_fields({
		"rewarded_ad_count_today": _rewarded_count_today,
		"rewarded_sub_counts": _rewarded_sub_counts,
		"last_ad_date": Time.get_date_string_from_system(),
	})


# ─── SDK Callbacks ───────────────────────────────────────
func _on_rewarded_ad_loaded() -> void:
	pass  # 광고 준비 완료


func _on_rewarded_ad_earned() -> void:
	_process_reward(_pending_reward_category)
	_pending_reward_category = ""
	_preload_rewarded_ad()  # 다음 광고 미리 로드


func _on_rewarded_ad_failed(error_code: int) -> void:
	rewarded_ad_failed.emit("sdk_error_%d" % error_code)
	_pending_reward_category = ""
	# 재시도 로직
	await get_tree().create_timer(5.0).timeout
	_preload_rewarded_ad()


func _on_interstitial_closed() -> void:
	interstitial_closed.emit()
	_preload_interstitial()


# ─── Preload ─────────────────────────────────────────────
func _preload_rewarded_ad() -> void:
	if _ad_plugin:
		_ad_plugin.call("loadRewardedAd")


func _preload_interstitial() -> void:
	if _ad_plugin:
		_ad_plugin.call("loadInterstitial")
```

```gdscript
# ─── Public API: 구매 ───────────────────────────────────
func purchase(product_id: String) -> void:
	## 구매 요청
	if product_id in _purchased_items:
		purchase_failed.emit(product_id, "already_owned")
		return

	_pending_purchase = product_id

	if _billing_plugin == null:
		# 디버그 환경: 즉시 지급
		_fulfill_purchase(product_id)
		return

	var os_name: String = OS.get_name()
	match os_name:
		"Android":
			_billing_plugin.call("purchase", product_id)
		"iOS":
			_billing_plugin.call("purchase", product_id)


func purchase_subscription(sub_id: String) -> void:
	## 구독 구매 요청
	_pending_purchase = sub_id
	if _billing_plugin:
		_billing_plugin.call("subscribe", sub_id)


func restore_purchases() -> void:
	## 구매 복원 (iOS 필수, Android 권장)
	if _billing_plugin:
		var os_name: String = OS.get_name()
		match os_name:
			"Android":
				_billing_plugin.call("queryPurchases", "inapp")
			"iOS":
				_billing_plugin.call("restorePurchases")


# ─── Public API: 상태 확인 ──────────────────────────────
func has_purchased(product_id: String) -> bool:
	return product_id in _purchased_items


func get_product_info(product_id: String) -> Dictionary:
	if product_id in PRODUCTS:
		return PRODUCTS[product_id].duplicate()
	if product_id in SUBSCRIPTIONS:
		return SUBSCRIPTIONS[product_id].duplicate()
	return {}


# ─── Internal: 구매 이행 ────────────────────────────────
func _fulfill_purchase(product_id: String) -> void:
	## 결제 완료 후 아이템 지급
	var product: Dictionary = PRODUCTS.get(product_id, {})
	if product.is_empty():
		# 구독 상품 처리
		_fulfill_subscription(product_id)
		return

	match product.get("type", ""):
		"non_consumable":
			_purchased_items.append(product_id)
			if product_id == "ad_remove":
				AdManager.set_ad_removal(true)

		"consumable":
			# 코인 지급
			if product.has("coins"):
				CoinManager.add_coins_from_iap(product["coins"])

	_save_receipt(product_id)
	_save_state()
	purchase_completed.emit(product_id)
	print("[IAPManager] 구매 완료: %s" % product_id)


func _fulfill_subscription(sub_id: String) -> void:
	## 구독 활성화
	var sub: Dictionary = SUBSCRIPTIONS.get(sub_id, {})
	if sub.is_empty():
		return

	var expiry: String = ""
	match sub.get("period", ""):
		"monthly":
			# 현재 시간 + 30일
			var unix: float = Time.get_unix_time_from_system() + (30 * 86400)
			expiry = Time.get_datetime_string_from_unix_time(int(unix))
		"yearly":
			# 현재 시간 + 365일
			var unix: float = Time.get_unix_time_from_system() + (365 * 86400)
			expiry = Time.get_datetime_string_from_unix_time(int(unix))

	SaveManager.set_fields({"subscription_expiry": expiry})
	SubscriptionManager.activate(expiry)
	AdManager.set_premium_status(true)
	purchase_completed.emit(sub_id)


# ─── Internal: 영수증 / 저장 ────────────────────────────
func _save_receipt(product_id: String) -> void:
	## 로컬 영수증 저장 (향후 서버 검증 전환 예정)
	var receipts: Array = SaveManager.get_data().get("purchase_receipts", [])
	receipts.append({
		"product_id": product_id,
		"timestamp": Time.get_unix_time_from_system(),
		"platform": OS.get_name(),
		# 실제 구현 시: 플랫폼 영수증 토큰 포함
		# "receipt_token": _billing_plugin.call("getReceiptToken"),
	})
	SaveManager.set_fields({"purchase_receipts": receipts})


func _save_state() -> void:
	SaveManager.set_fields({
		"purchased_items": _purchased_items,
	})


# ─── Billing SDK Callbacks ──────────────────────────────
func _on_billing_connected() -> void:
	print("[IAPManager] Billing 연결 완료")
	# 상품 정보 쿼리
	var product_ids: Array = PRODUCTS.keys() + SUBSCRIPTIONS.keys()
	_billing_plugin.call("querySkuDetails", product_ids, "inapp")


func _on_billing_disconnected() -> void:
	push_warning("[IAPManager] Billing 연결 끊김, 재연결 시도")
	await get_tree().create_timer(3.0).timeout
	_billing_plugin.call("startConnection")


func _on_purchases_updated(purchases: Array) -> void:
	for purchase_data in purchases:
		var product_id: String = purchase_data.get("sku", "")
		var purchase_state: int = purchase_data.get("purchase_state", -1)
		if purchase_state == 1:  # PURCHASED
			# Acknowledge 또는 Consume
			var product: Dictionary = PRODUCTS.get(product_id, {})
			if product.get("type", "") == "non_consumable":
				_billing_plugin.call("acknowledgePurchase",
					purchase_data.get("purchase_token", ""))
			else:
				_billing_plugin.call("consumePurchase",
					purchase_data.get("purchase_token", ""))


func _on_purchase_acknowledged(purchase_token: String) -> void:
	_fulfill_purchase(_pending_purchase)
	_pending_purchase = ""


func _on_purchase_consumed(purchase_token: String) -> void:
	_fulfill_purchase(_pending_purchase)
	_pending_purchase = ""


func _on_purchase_error(error_code: int, error_message: String) -> void:
	purchase_failed.emit(_pending_purchase, error_message)
	_pending_purchase = ""


func _on_query_purchases(purchases: Array) -> void:
	## 구매 복원 처리
	for purchase_data in purchases:
		var product_id: String = purchase_data.get("sku", "")
		if product_id in PRODUCTS:
			if PRODUCTS[product_id].get("type") == "non_consumable":
				if product_id not in _purchased_items:
					_purchased_items.append(product_id)
					if product_id == "ad_remove":
						AdManager.set_ad_removal(true)
					purchase_restored.emit(product_id)
	_save_state()


func _on_storekit_transaction(transaction: Dictionary) -> void:
	var product_id: String = transaction.get("product_id", "")
	var state: String = transaction.get("state", "")
	match state:
		"purchased", "restored":
			_fulfill_purchase(product_id)
		"failed":
			purchase_failed.emit(product_id, transaction.get("error", "unknown"))


func _on_storekit_restore(results: Array) -> void:
	for product_id in results:
		if product_id not in _purchased_items:
			_purchased_items.append(product_id)
			purchase_restored.emit(product_id)
	_save_state()
```

### 7.3 구독 상태 관리

```gdscript
## subscription_manager.gd (개요)
## 구독 상태 확인 및 혜택 적용 관리
extends Node

signal subscription_status_changed(is_active: bool)

var _expiry_timestamp: float = 0.0
var _is_active: bool = false
var _has_free_trial: bool = true


func _ready() -> void:
	_load_state()
	_check_expiry()


func _load_state() -> void:
	var save_data: Dictionary = SaveManager.get_data()
	var expiry_str: String = save_data.get("subscription_expiry", "")
	if not expiry_str.is_empty():
		# ISO 8601 파싱
		var datetime: Dictionary = Time.get_datetime_dict_from_datetime_string(expiry_str, false)
		_expiry_timestamp = Time.get_unix_time_from_datetime_dict(datetime)
	_has_free_trial = save_data.get("has_free_trial", true)


func _check_expiry() -> void:
	var now: float = Time.get_unix_time_from_system()
	var was_active: bool = _is_active
	_is_active = now < _expiry_timestamp
	if was_active != _is_active:
		subscription_status_changed.emit(_is_active)
		if not _is_active:
			AdManager.set_premium_status(false)


func is_premium_active() -> bool:
	_check_expiry()
	return _is_active


func activate(expiry_iso: String) -> void:
	var datetime: Dictionary = Time.get_datetime_dict_from_datetime_string(expiry_iso, false)
	_expiry_timestamp = Time.get_unix_time_from_datetime_dict(datetime)
	_is_active = true
	_has_free_trial = false
	SaveManager.set_fields({
		"subscription_expiry": expiry_iso,
		"has_free_trial": false,
	})
	subscription_status_changed.emit(true)
	AdManager.set_premium_status(true)


func get_expiry_string() -> String:
	if _expiry_timestamp <= 0.0:
		return ""
	return Time.get_datetime_string_from_unix_time(int(_expiry_timestamp))


func has_free_trial_available() -> bool:
	return _has_free_trial
```

> **밸런스 핵심**: 초~중급 구간에서 코인을 축적하게 하고, 고급 이상에서 자연스럽게 소비가 늘어나도록 설계. 과금 없이도 클리어 가능하되, 편의를 위한 과금 동기 부여.

#### 로컬 검증 (Phase 1)

```gdscript
## receipt_validator.gd
## 결제 영수증 로컬 검증 (Phase 1)
## 향후 서버 검증으로 전환 예정

class_name ReceiptValidator


static func validate_local(receipt: Dictionary) -> bool:
	## 기본 무결성 체크
	if not receipt.has("product_id"):
		return false
	if not receipt.has("timestamp"):
		return false
	if not receipt.has("platform"):
		return false

	# 타임스탬프 유효성 (미래 시간 방지)
	var now: float = Time.get_unix_time_from_system()
	var receipt_time: float = receipt.get("timestamp", 0.0)
	if receipt_time > now + 3600:  # 1시간 이상 미래
		return false

	# 중복 구매 체크 (같은 product_id + 같은 timestamp)
	var receipts: Array = SaveManager.get_data().get("purchase_receipts", [])
	for existing in receipts:
		if existing.get("product_id") == receipt["product_id"] \
			and existing.get("timestamp") == receipt["timestamp"]:
			return false  # 중복

	return true
```

## 11. 윤리적 수익화 원칙

### 11.1 P2W(Pay-to-Win) 방지

본 게임은 **Word Search 퍼즐**이라는 장르 특성상, 과금이 게임 실력이나 경쟁 결과에 직접적인 영향을 주지 않도록 설계한다.

| 원칙 | 적용 |
|------|------|
| **힌트는 편의 기능** | 힌트 사용 시 등급이 낮아짐 (S등급 조건: 노힌트) |
| **리더보드 공정성** | 리더보드 점수는 시간 + 노힌트 기반, 과금 무관 |
| **콘텐츠 잠금 없음** | 모든 스테이지는 순서대로 플레이 가능, 과금 필요 없음 |
| **시간 벽 없음** | 에너지/스태미나 시스템 미적용 |
| **과금 전용 파워업 없음** | IAP 전용 아이템은 커스터마이제이션(외형)뿐 |
